34 /*
35  * The s390 Performance Measurement counter facility does not have a fixed
36  * type number anymore. This was caused by linux kernel commits
37  * 66d258c5b0488 perf/core: Optimize perf_init_event()
38  * and its necessary follow on commit
39  * 6a82e23f45fe0 s390/cpumf: Adjust registration of s390 PMU device drivers
40  *
41  * Now read out the current type number from a sysfs file named
42  * /sys/devices/cpum_cf/type. If it does not exist there is no CPU-MF counter
43  * facility installed or activated.
44  *
45  * As the CPU Measurement counter facility does not change on a running
46  * system, read out the type value on first read and cache it.
47  *
48  * There are several PMUs for s390, so find the correct one first and return
49  * its PMU type value assigned at system boot time.
50  */
51 static struct pfm_s390_perf_aptt { /* Perf attribute PMU type table */
52  pfm_pmu_t pmutype; /* PMU Type number */
53  const char *fname; /* File name to read type from */
54  unsigned int value; /* Type value, 0 --> unused */
55 } pfm_s390_perf_aptt[] = {
56  {
57  .pmutype = PFM_PMU_S390X_CPUM_CF,
58  .fname = "/sys/bus/event_source/devices/cpum_cf/type"
59  },
60  {
61  .pmutype = PFM_PMU_S390X_CPUM_SF,
62  .fname = "/sys/bus/event_source/devices/cpum_sf/type"
63  },
64 };
65 #define S390_APTT_COUNT LIBPFM_ARRAY_SIZE(pfm_s390_perf_aptt)

67 static int pfm_s390_get_perf_attr_type(pfm_pmu_t pmutype)
68 {
69  int cpum_cf_type;
70  size_t buflen;
71  char *buffer;
72  FILE *fp;
73  size_t i;
74 
75  /* Find type of PMU and return known and cached value */
76  for (i = 0; i < S390_APTT_COUNT; ++i) {
77  if (pfm_s390_perf_aptt[i].pmutype == pmutype)
78  break;
79  }
80 
81  if (i == S390_APTT_COUNT)
82  return PFM_ERR_NOTFOUND;
83 
84  if (pfm_s390_perf_aptt[i].value)
85  return pfm_s390_perf_aptt[i].value;
86 
87  /* Value unknown, read from file */
88  fp = fopen(pfm_s390_perf_aptt[i].fname, "r");
89  if (fp == NULL)
90  return PFM_ERR_NOTFOUND;
91 
92  buffer = NULL;
93 
94  if (pfmlib_getl(&buffer, &buflen, fp) != -1 &&
95  sscanf(buffer, "%u", &cpum_cf_type) == -1)
96  cpum_cf_type = PERF_TYPE_RAW;
97 
98  fclose(fp);
99  free(buffer);
100 
101  pfm_s390_perf_aptt[i].value = cpum_cf_type;
102  return cpum_cf_type;
103 }

105 int pfm_s390x_get_perf_encoding(void *this, pfmlib_event_desc_t *e)
106 {
107  pfmlib_pmu_t *pmu = this;
108  struct perf_event_attr *attr = e->os_data;
109  int rc;
110 
111  if (!pmu->get_event_encoding[PFM_OS_NONE])
112  return PFM_ERR_NOTSUPP;
113 
114  /* set up raw pmu event encoding */
115  rc = pmu->get_event_encoding[PFM_OS_NONE](this, e);
116  if (rc == PFM_SUCCESS) {
117  /* currently use raw events only */
118  rc = pfm_s390_get_perf_attr_type(pmu->pmu);
119  if (rc > 0) { /* PMU types are positive */
120  attr->type = rc;
121  attr->config = e->codes[0];
122  rc = PFM_SUCCESS;
123  }
124  }
125 
126  return rc;
127 }

146  }

150  compact = 1;
151

155  }
