78 static int
79 pfmlib_perf_event_encode(void *this, const char *str, int dfl_plm, void *data)
80 {
81  pfm_perf_encode_arg_t arg;
82  pfm_perf_encode_arg_t *uarg = data;
83  pfmlib_os_t *os = this;
84  struct perf_event_attr my_attr, *attr;
85  pfmlib_pmu_t *pmu;
86  pfmlib_event_desc_t e;
87  pfmlib_event_attr_info_t *a;
88  size_t orig_sz, asz, sz = sizeof(arg);
89  uint64_t ival;
90  int has_plm = 0, has_vmx_plm = 0;
91  int i, plm = 0, ret, vmx_plm = 0;
92  int cpu = -1, pinned = 0;
93 
94  sz = pfmlib_check_struct(uarg, uarg->size, PFM_PERF_ENCODE_ABI0, sz);
95  if (!sz)
96  return PFM_ERR_INVAL;
97 
98  /* copy input */
99  memcpy(&arg, uarg, sz);
100 
101  /* pointer to our internal attr struct */
102  memset(&my_attr, 0, sizeof(my_attr));
103  attr = &my_attr;
104 
105  /*
106  * copy user attr to our internal version
107  * size == 0 is interpreted minimal possible
108  * size (ABI_VER0)
109  */
110 
111  /* size of attr struct passed by user */
112  orig_sz = uarg->attr->size;
113 
114  if (orig_sz == 0)
115  asz = PERF_ATTR_SIZE_VER0;
116  else
117  asz = min(sizeof(*attr), orig_sz);
118 
119  /*
120  * we copy the user struct to preserve whatever may
121  * have been initialized but that we do not use
122  */
123  memcpy(attr, uarg->attr, asz);
124 
125  /* restore internal size (just in case we need it) */
126  attr->size = sizeof(my_attr);
127 
128  /* useful for debugging */
129  if (asz != sizeof(*attr))
130  __pfm_vbprintf("warning: mismatch attr struct size "
131  "user=%d libpfm=%zu\n", asz, sizeof(*attr));
132 
133  memset(&e, 0, sizeof(e));
134 
135  e.osid = os->id;
136  e.os_data = attr;
137  e.dfl_plm = dfl_plm;
138 
139  /* after this call, need to call pfmlib_release_event() */
140  ret = pfmlib_parse_event(str, &e);
141  if (ret != PFM_SUCCESS)
142  return ret;
143 
144  pmu = e.pmu;
145 
146  ret = PFM_ERR_NOTSUPP;
147  if (!pmu->get_event_encoding[e.osid]) {
148  DPRINT("PMU %s does not support PFM_OS_NONE\n", pmu->name);
149  goto done;
150  }
151 
152  ret = pmu->get_event_encoding[e.osid](pmu, &e);
153  if (ret != PFM_SUCCESS)
154  goto done;
155 
156  /*
157  * process perf_event attributes
158  */
159  for (i = 0; i < e.nattrs; i++) {
160 
161  a = attr(&e, i);
162 
163  if (a->ctrl != PFM_ATTR_CTRL_PERF_EVENT)
164  continue;
165 
166  ival = e.attrs[i].ival;
167 
168  switch(a->idx) {
169  case PERF_ATTR_U:
170  if (ival)
171  plm |= PFM_PLM3;
172  has_plm = 1;
173  break;
174  case PERF_ATTR_K:
175  if (ival)
176  plm |= PFM_PLM0;
177  has_plm = 1;
178  break;
179  case PERF_ATTR_H:
180  if (ival)
181  plm |= PFM_PLMH;
182  has_plm = 1;
183  break;
184  case PERF_ATTR_PE:
185  if (!ival || attr->freq) {
186  ret = PFM_ERR_ATTR_VAL;
187  goto done;
188  }
189  attr->sample_period = ival;
190  break;
191  case PERF_ATTR_FR:
192  if (!ival || attr->sample_period) {
193  ret = PFM_ERR_ATTR_VAL;
194  goto done;
195  }
196  attr->sample_freq = ival;
197  attr->freq = 1;
198  break;
199  case PERF_ATTR_PR:
200  if (ival > 3) {
201  ret = PFM_ERR_ATTR_VAL;
202  goto done;
203  }
204  attr->precise_ip = ival;
205  break;
206  case PERF_ATTR_EX:
207  if (ival && !attr->exclusive)
208  attr->exclusive = 1;
209  break;
210  case PERF_ATTR_MG:
211  vmx_plm |= PFM_PLM3;
212  has_vmx_plm = 1;
213  break;
214  case PERF_ATTR_MH:
215  vmx_plm |= PFM_PLM0;
216  has_vmx_plm = 1;
217  break;
218  case PERF_ATTR_CPU:
219  if (ival >= INT_MAX) {
220  ret = PFM_ERR_ATTR_VAL;
221  goto done;
222  }
223  cpu = (int)ival;
224  break;
225  case PERF_ATTR_PIN:
226  pinned = (int)!!ival;
227  break;
228  case PERF_ATTR_HWS:
229  attr->precise_ip = (int)!!ival;
230  break;
231  }
232  }
233  /*
234  * if no priv level mask was provided
235  * with the event, then use dfl_plm
236  */
237  if (!has_plm)
238  plm = dfl_plm;
239 
240  /* exclude_guest by default */
241  if (!has_vmx_plm)
242  vmx_plm = PFM_PLM0;
243  /*
244  * perf_event plm work by exclusion, so use logical or
245  * goal here is to set to zero any exclude_* not supported
246  * by underlying PMU
247  */
248  plm |= (~pmu->supported_plm) & PFM_PLM_ALL;
249  vmx_plm |= (~pmu->supported_plm) & PFM_PLM_ALL;
250 
251  attr->exclude_user = !(plm & PFM_PLM3);
252  attr->exclude_kernel = !(plm & PFM_PLM0);
253  attr->exclude_hv = !(plm & PFM_PLMH);
254  attr->exclude_guest = !(vmx_plm & PFM_PLM3);
255  attr->exclude_host = !(vmx_plm & PFM_PLM0);
256  attr->pinned = pinned;
257 
258  __pfm_vbprintf("PERF[type=%x config=0x%"PRIx64" config1=0x%"PRIx64
259  " excl=%d excl_user=%d excl_kernel=%d excl_hv=%d excl_host=%d excl_guest=%d period=%"PRIu64" freq=%d"
260  " precise=%d pinned=%d] %s\n",
261  attr->type,
262  attr->config,
263  attr->config1,
264  attr->exclusive,
265  attr->exclude_user,
266  attr->exclude_kernel,
267  attr->exclude_hv,
268  attr->exclude_host,
269  attr->exclude_guest,
270  attr->sample_period,
271  attr->freq,
272  attr->precise_ip,
273  attr->pinned,
274  str);
275 
276  /*
277  * propagate event index if necessary
278  */
279  arg.idx = pfmlib_pidx2idx(e.pmu, e.event);
280 
281  /* propagate cpu */
282  arg.cpu = cpu;
283 
284  /* propagate our changes, that overwrites attr->size */
285  memcpy(uarg->attr, attr, asz);
286 
287  /* restore user size */
288  uarg->attr->size = orig_sz;
289 
290  /*
291  * fstr not requested, stop here
292  */
293  ret = PFM_SUCCESS;
294  if (!arg.fstr) {
295  memcpy(uarg, &arg, sz);
296  goto done;
297  }
298 
299  for (i=0; i < e.npattrs; i++) {
300  int idx;
301 
302  if (e.pattrs[i].ctrl != PFM_ATTR_CTRL_PERF_EVENT)
303  continue;
304 
305  idx = e.pattrs[i].idx;
306  switch (idx) {
307  case PERF_ATTR_K:
308  evt_strcat(e.fstr, ":%s=%lu", perf_event_ext_mods[idx].name, !!(plm & PFM_PLM0));
309  break;
310  case PERF_ATTR_U:
311  evt_strcat(e.fstr, ":%s=%lu", perf_event_ext_mods[idx].name, !!(plm & PFM_PLM3));
312  break;
313  case PERF_ATTR_H:
314  evt_strcat(e.fstr, ":%s=%lu", perf_event_ext_mods[idx].name, !!(plm & PFM_PLMH));
315  break;
316  case PERF_ATTR_PR:
317  case PERF_ATTR_HWS:
318  evt_strcat(e.fstr, ":%s=%d", perf_event_ext_mods[idx].name, attr->precise_ip);
319  break;
320  case PERF_ATTR_PE:
321  case PERF_ATTR_FR:
322  if (attr->freq && attr->sample_period)
323  evt_strcat(e.fstr, ":%s=%"PRIu64, perf_event_ext_mods[idx].name, attr->sample_period);
324  else if (attr->sample_period)
325  evt_strcat(e.fstr, ":%s=%"PRIu64, perf_event_ext_mods[idx].name, attr->sample_period);
326  break;
327  case PERF_ATTR_MG:
328  evt_strcat(e.fstr, ":%s=%lu", perf_event_ext_mods[idx].name, !attr->exclude_guest);
329  break;
330  case PERF_ATTR_MH:
331  evt_strcat(e.fstr, ":%s=%lu", perf_event_ext_mods[idx].name, !attr->exclude_host);
332  break;
333  case PERF_ATTR_EX:
334  evt_strcat(e.fstr, ":%s=%lu", perf_event_ext_mods[idx].name, attr->exclusive);
335  break;
336  }
337  }
338 
339  ret = pfmlib_build_fstr(&e, arg.fstr);
340  if (ret == PFM_SUCCESS)
341  memcpy(uarg, &arg, sz);
342 
343 done:
344  pfmlib_release_event(&e);
345  return ret;
346 }

362 static int
363 perf_get_os_attr_info(void *this, pfmlib_event_desc_t *e)
364 {
365  pfmlib_os_t *os = this;
366  pfmlib_event_attr_info_t *info;
367  int i, k, j = e->npattrs;
368 
369  for (i = k = 0; os->atdesc[i].name; i++) {
370  /* skip padding entries */
371  if (is_empty_attr(os->atdesc+i))
372  continue;
373 
374  info = e->pattrs + j + k;
375 
376  info->name = os->atdesc[i].name;
377  info->desc = os->atdesc[i].desc;
378  info->equiv= NULL;
379  info->code = i;
380  info->idx = i; /* namespace-specific index */
381  info->type = os->atdesc[i].type;
382  info->is_dfl = 0;
383  info->ctrl = PFM_ATTR_CTRL_PERF_EVENT;
384  k++;
385  }
386  e->npattrs += k;
387 
388  return PFM_SUCCESS;
389 }
