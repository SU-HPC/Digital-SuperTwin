248 {
249  const intel_x86_entry_t *pe = this_pe(this);
250 
251  filters[0].val |= pe[event].umasks[umask].ufilters[0] & ((1ULL << 32)-1);
252  filters[0].val &= ~(pe[event].umasks[umask].ufilters[0] >> 32);
253 
254  filters[1].val |= pe[event].umasks[umask].ufilters[1] & ((1ULL << 32)-1);
255  filters[1].val &= ~(pe[event].umasks[umask].ufilters[1] >>32);

258 int
259 snbep_unc_add_defaults(void *this, pfmlib_event_desc_t *e,
260  unsigned int msk,
261  uint64_t *umask,
262  pfm_snbep_unc_reg_t *filters,
263  unsigned short max_grpid,
264  int *numasks)
265 {
266  const intel_x86_entry_t *pe = this_pe(this);
267  const intel_x86_entry_t *ent;
268  unsigned int i;
269  int j, k, added, skip;
270  int idx;
271 
272  k = e->nattrs;
273  ent = pe+e->event;
274 
275  for(i=0; msk; msk >>=1, i++) {
276 
277  if (!(msk & 0x1))
278  continue;
279 
280  added = skip = 0;
281 
282  for (j = 0; j < e->npattrs; j++) {
283  if (e->pattrs[j].ctrl != PFM_ATTR_CTRL_PMU)
284  continue;
285 
286  if (e->pattrs[j].type != PFM_ATTR_UMASK)
287  continue;
288 
289  idx = e->pattrs[j].idx;
290 
291  if (get_grpid(ent->umasks[idx].grpid) != i)
292  continue;
293 
294  if (max_grpid != INTEL_X86_MAX_GRPID && i > max_grpid) {
295  skip = 1;
296  continue;
297  }
298 
299  if (intel_x86_uflag(this, e->event, idx, INTEL_X86_GRP_DFL_NONE)) {
300  skip = 1;
301  continue;
302  }
303 
304  /* umask is default for group */
305  if (intel_x86_uflag(this, e->event, idx, INTEL_X86_DFL)) {
306  DPRINT("added default %s for group %d j=%d idx=%d ucode=0x%"PRIx64"\n",
307  ent->umasks[idx].uname,
308  i,
309  j,
310  idx,
311  ent->umasks[idx].ucode);
312  /*
313  * default could be an alias, but
314  * ucode must reflect actual code
315  */
316  *umask |= ent->umasks[idx].ucode >> 8;
317 
318  set_filters(this, filters, e->event, idx);
319 
320  e->attrs[k].id = j; /* pattrs index */
321  e->attrs[k].ival = 0;
322  k++;
323 
324  (*numasks)++;
325  added++;
326  if (intel_x86_eflag(this, e->event, INTEL_X86_GRP_EXCL))
327  goto done;
328 
329  if (intel_x86_uflag(this, e->event, idx, INTEL_X86_EXCL_GRP_GT)) {
330  if (max_grpid != INTEL_X86_MAX_GRPID) {
331  DPRINT("two max_grpid, old=%d new=%d\n", max_grpid, get_grpid(ent->umasks[idx].grpid));
332  return PFM_ERR_UMASK;
333  }
334  max_grpid = get_grpid(ent->umasks[idx].grpid);
335  }
336  }
337  }
338  if (!added && !skip) {
339  DPRINT("no default found for event %s unit mask group %d (max_grpid=%d, i=%d)\n", ent->name, i, max_grpid, i);
340  return PFM_ERR_UMASK;
341  }
342  }
343  DPRINT("max_grpid=%d nattrs=%d k=%d umask=0x%"PRIx64"\n", max_grpid, e->nattrs, k, *umask);
344 done:
345  e->nattrs = k;
346  return PFM_SUCCESS;
347 }

351  * common encoding routine
352  */
353 int
354 pfm_intel_snbep_unc_get_encoding(void *this, pfmlib_event_desc_t *e)
355 {
356  const intel_x86_entry_t *pe = this_pe(this);
357  unsigned int grpmsk, ugrpmsk = 0;
358  unsigned short max_grpid = INTEL_X86_MAX_GRPID;
359  unsigned short last_grpid = INTEL_X86_MAX_GRPID;
360  unsigned short req_grpid;
361  int umodmsk = 0, modmsk_r = 0;
362  int pcu_filt_band = -1;
363  pfm_snbep_unc_reg_t reg;
364  pfm_snbep_unc_reg_t filters[INTEL_X86_MAX_FILTERS];
365  pfm_snbep_unc_reg_t addr;
366  pfmlib_event_attr_info_t *a;
367  uint64_t val, umask1, umask2;
368  int k, ret, numasks = 0;
369  int must_have_filt0 = 0;
370  int max_req_grpid = -1;
371  unsigned short grpid;
372  int grpcounts[INTEL_X86_NUM_GRP];
373  int req_grps[INTEL_X86_NUM_GRP];
374  int ncombo[INTEL_X86_NUM_GRP];
375  char umask_str[PFMLIB_EVT_MAX_NAME_LEN];
376 
377  memset(grpcounts, 0, sizeof(grpcounts));
378  memset(ncombo, 0, sizeof(ncombo));
379  memset(filters, 0, sizeof(filters));
380 
381  addr.val = 0;
382 
383  umask_str[0] = e->fstr[0] = '\0';
384 
385  reg.val = val = pe[e->event].code;
386 
387  /* take into account hardcoded umask */
388  umask1 = (val >> 8);
389  umask2 = umask1;
390 
391  grpmsk = (1 << pe[e->event].ngrp)-1;
392 
393  modmsk_r = pe[e->event].modmsk_req;
394 
395  if (intel_x86_eflag(this, e->event, INTEL_X86_FORCE_FILT0))
396  must_have_filt0 = 1;
397 
398  for(k=0; k < e->nattrs; k++) {
399  a = attr(e, k);
400 
401  if (a->ctrl != PFM_ATTR_CTRL_PMU)
402  continue;
403 
404  if (a->type == PFM_ATTR_UMASK) {
405  uint64_t um;
406 
407  grpid = get_grpid(pe[e->event].umasks[a->idx].grpid);
408  req_grpid = get_req_grpid(pe[e->event].umasks[a->idx].grpid);
409 
410  /*
411  * certain event groups are meant to be
412  * exclusive, i.e., only unit masks of one group
413  * can be used
414  */
415  if (last_grpid != INTEL_X86_MAX_GRPID && grpid != last_grpid
416  && intel_x86_eflag(this, e->event, INTEL_X86_GRP_EXCL)) {
417  DPRINT("exclusive unit mask group error\n");
418  return PFM_ERR_FEATCOMB;
419  }
420 
421  /*
422  * selecting certain umasks in a group may exclude any umasks
423  * from any groups with a higher index
424  *
425  * enforcement requires looking at the grpid of all the umasks
426  */
427  if (intel_x86_uflag(this, e->event, a->idx, INTEL_X86_EXCL_GRP_GT))
428  max_grpid = grpid;
429 
430  /*
431  * certain event groups are meant to be
432  * exclusive, i.e., only unit masks of one group
433  * can be used
434  */
435  if (last_grpid != INTEL_X86_MAX_GRPID && grpid != last_grpid
436  && intel_x86_eflag(this, e->event, INTEL_X86_GRP_EXCL)) {
437  DPRINT("exclusive unit mask group error\n");
438  return PFM_ERR_FEATCOMB;
439  }
440  /*
441  * upper layer has removed duplicates
442  * so if we come here more than once, it is for two
443  * disinct umasks
444  *
445  * NCOMBO=no combination of unit masks within the same
446  * umask group
447  */
448  ++grpcounts[grpid];
449  if (intel_x86_uflag(this, e->event, a->idx, INTEL_X86_GRP_REQ)) {
450  DPRINT("event requires grpid %d\n", req_grpid);
451  /* initialize req_grpcounts array only when needed */
452  if (max_req_grpid == -1) {
453  int x;
454  for (x = 0; x < INTEL_X86_NUM_GRP; x++)
455  req_grps[x] = 0xff;
456  }
457  if (req_grpid > max_req_grpid)
458  max_req_grpid = req_grpid;
459  DPRINT("max_req_grpid=%d\n", max_req_grpid);
460  req_grps[req_grpid] = 1;
461  }
462 
463  /* mark that we have a umask with NCOMBO in this group */
464  if (intel_x86_uflag(this, e->event, a->idx, INTEL_X86_NCOMBO))
465  ncombo[grpid] = 1;
466 
467  /*
468  * if more than one umask in this group but one is marked
469  * with ncombo, then fail. It is okay to combine umask within
470  * a group as long as none is tagged with NCOMBO
471  */
472  if (grpcounts[grpid] > 1 && ncombo[grpid]) {
473  DPRINT("umask %s does not support unit mask combination within group %d\n", pe[e->event].umasks[a->idx].uname, grpid);
474  return PFM_ERR_FEATCOMB;
475  }
476 
477  last_grpid = grpid;
478 
479  um = pe[e->event].umasks[a->idx].ucode;
480 
481  set_filters(this, filters, e->event, a->idx);
482 
483  um >>= 8;
484  umask2 |= um;
485 
486  ugrpmsk |= 1 << grpid;
487 
488  /* PCU occ event */
489  if (is_occ_event(this, e->event)) {
490  reg.pcu.unc_occ = umask2 >> 6;
491  umask2 = 0;
492  } else
493  reg.val |= umask2 << 8;
494 
495  evt_strcat(umask_str, ":%s", pe[e->event].umasks[a->idx].uname);
496 
497  modmsk_r |= pe[e->event].umasks[a->idx].umodmsk_req;
498  numasks++;
499 
500  } else if (a->type == PFM_ATTR_RAW_UMASK) {
501 
502  /* there can only be one RAW_UMASK per event */
503 
504  /* sanity check */
505  if (a->idx & ~0xff) {
506  DPRINT("raw umask is 8-bit wide\n");
507  return PFM_ERR_ATTR;
508  }
509  /* override umask */
510  umask2 = a->idx & 0xff;
511  ugrpmsk = grpmsk;
512  numasks++;
513  } else {
514  uint64_t ival = e->attrs[k].ival;
515  switch(a->idx) {
516  case SNBEP_UNC_ATTR_I: /* invert */
517  if (is_occ_event(this, e->event))
518  reg.pcu.unc_occ_inv = !!ival;
519  else
520  reg.com.unc_inv = !!ival;
521  umodmsk |= _SNBEP_UNC_ATTR_I;
522  break;
523  case SNBEP_UNC_ATTR_E: /* edge */
524  if (is_occ_event(this, e->event))
525  reg.pcu.unc_occ_edge = !!ival;
526  else
527  reg.com.unc_edge = !!ival;
528  umodmsk |= _SNBEP_UNC_ATTR_E;
529  break;
530  case SNBEP_UNC_ATTR_T8: /* counter-mask */
531  /* already forced, cannot overwrite */
532  if (ival > 255)
533  return PFM_ERR_ATTR_VAL;
534  reg.com.unc_thres = ival;
535  umodmsk |= _SNBEP_UNC_ATTR_T8;
536  break;
537  case SNBEP_UNC_ATTR_T5: /* pcu counter-mask */
538  /* already forced, cannot overwrite */
539  if (ival > 31)
540  return PFM_ERR_ATTR_VAL;
541  reg.pcu.unc_thres = ival;
542  umodmsk |= _SNBEP_UNC_ATTR_T5;
543  break;
544  case SNBEP_UNC_ATTR_TF: /* thread id */
545  if (ival > 1) {
546  DPRINT("invalid thread id, must be < 1");
547  return PFM_ERR_ATTR_VAL;
548  }
549  reg.cbo.unc_tid = 1;
550  must_have_filt0 = 1;
551  filters[0].cbo_filt.tid = ival;
552  umodmsk |= _SNBEP_UNC_ATTR_TF;
553  break;
554  case SNBEP_UNC_ATTR_TF1: /* thread id skx */
555  if (ival > 7)
556  return PFM_ERR_ATTR_VAL;
557  reg.cha.unc_tid = 1;
558  filters[0].skx_cha_filt0.tid = ival; /* includes non-thread data */
559  must_have_filt0 = 1;
560  umodmsk |= _SNBEP_UNC_ATTR_TF1;
561  break;
562  case SNBEP_UNC_ATTR_CF: /* core id */
563  if (ival > 15)
564  return PFM_ERR_ATTR_VAL;
565  reg.cbo.unc_tid = 1;
566  filters[0].cbo_filt.cid = ival;
567  must_have_filt0 = 1;
568  umodmsk |= _SNBEP_UNC_ATTR_CF;
569  break;
570  case SNBEP_UNC_ATTR_CF1: /* core id */
571  if (ival > 63)
572  return PFM_ERR_ATTR_VAL;
573  reg.cbo.unc_tid = 1;
574  filters[0].hswep_cbo_filt0.cid = ival; /* includes non-thread data */
575  must_have_filt0 = 1;
576  umodmsk |= _SNBEP_UNC_ATTR_CF1;
577  break;
578  case SNBEP_UNC_ATTR_NF: /* node id filter0 */
579  if (ival > 255 || ival == 0) {
580  DPRINT("invalid nf, 0 < nf < 256\n");
581  return PFM_ERR_ATTR_VAL;
582  }
583  filters[0].cbo_filt.nid = ival;
584  umodmsk |= _SNBEP_UNC_ATTR_NF;
585  break;
586  case SNBEP_UNC_ATTR_NF1: /* node id filter1 */
587  if (ival > 255 || ival == 0) {
588  DPRINT("invalid nf, 0 < nf < 256\n");
589  return PFM_ERR_ATTR_VAL;
590  }
591  filters[1].ivbep_cbo_filt1.nid = ival;
592  umodmsk |= _SNBEP_UNC_ATTR_NF1;
593  break;
594  case SNBEP_UNC_ATTR_CF2: /* src-id/core-id skx */
595  if (ival > 64)
596  return PFM_ERR_ATTR_VAL;
597  reg.cha.unc_tid = 1;
598  filters[0].skx_cha_filt0.sid = ival;
599  must_have_filt0 = 1;
600  umodmsk |= _SNBEP_UNC_ATTR_CF2;
601  break;
602  case SNBEP_UNC_ATTR_FF: /* freq band filter */
603  if (ival > 255)
604  return PFM_ERR_ATTR_VAL;
