218 template <
219  typename AIteratorT,
220  typename BIteratorT,
221  typename OffsetT,
222  typename CoordinateT>
223 inline void MergePathSearch(
224  OffsetT diagonal,
225  AIteratorT a,
226  BIteratorT b,
227  OffsetT a_len,
228  OffsetT b_len,
229  CoordinateT& path_coordinate)
230 {
231  OffsetT x_min = std::max(diagonal - b_len, 0);
232  OffsetT x_max = std::min(diagonal, a_len);
233 
234  while (x_min < x_max)
235  {
236  OffsetT x_pivot = (x_min + x_max) >> 1;
237  if (a[x_pivot] <= b[diagonal - x_pivot - 1])
238  x_min = x_pivot + 1; // Contract range up A (down B)
239  else
240  x_max = x_pivot; // Contract range down A (up B)
241  }
242 
243  path_coordinate.x = std::min(x_min, a_len);
244  path_coordinate.y = diagonal - x_min;
245 }

253 // Compute reference SpMV y = Ax
254 template <
255  typename ValueT,
256  typename OffsetT>
257 void SpmvGold(
258  CsrMatrix<ValueT, OffsetT>& a,
259  ValueT* vector_x,
260  ValueT* vector_y_in,
261  ValueT* vector_y_out,
262  ValueT alpha,
263  ValueT beta)
264 {
265  for (OffsetT row = 0; row < a.num_rows; ++row)
266  {
267  ValueT partial = beta * vector_y_in[row];
268  for (
269  OffsetT offset = a.row_offsets[row];
270  offset < a.row_offsets[row + 1];
271  ++offset)
272  {
273  partial += alpha * a.values[offset] * vector_x[a.column_indices[offset]];
274  }
275  vector_y_out[row] = partial;
276  }
277 }

289 template <
290  typename ValueT,
291  typename OffsetT>
292 void OmpMergeCsrmv(
293  int num_threads,
294  CsrMatrix<ValueT, OffsetT>& a,
295  OffsetT* __restrict row_end_offsets,
296  OffsetT* __restrict column_indices,
297  ValueT* __restrict values,
298  ValueT* __restrict vector_x,
299  ValueT* __restrict vector_y_out)
300 {
301  // Temporary storage for inter-thread fix-up after load-balanced work
302  OffsetT row_carry_out[256]; // The last row-id each worked on by each thread when it finished its path segment
303  ValueT value_carry_out[256]; // The running total within each thread when it finished its path segment
304 
305  #pragma omp parallel for schedule(static) num_threads(num_threads)
306  for (int tid = 0; tid < num_threads; tid++)
307  {
308  // Merge list B (NZ indices)
309  CountingInputIterator<OffsetT> nonzero_indices(0);
310 
311  OffsetT num_merge_items = a.num_rows + a.num_nonzeros; // Merge path total length
312  OffsetT items_per_thread = (num_merge_items + num_threads - 1) / num_threads; // Merge items per thread
313 
314  // Find starting and ending MergePath coordinates (row-idx, nonzero-idx) for each thread
315  int2 thread_coord;
316  int2 thread_coord_end;
317  int start_diagonal = std::min(items_per_thread * tid, num_merge_items);
318  int end_diagonal = std::min(start_diagonal + items_per_thread, num_merge_items);
319 
320  MergePathSearch(start_diagonal, row_end_offsets, nonzero_indices, a.num_rows, a.num_nonzeros, thread_coord);
321  MergePathSearch(end_diagonal, row_end_offsets, nonzero_indices, a.num_rows, a.num_nonzeros, thread_coord_end);
322 
323  // Consume whole rows
324  for (; thread_coord.x < thread_coord_end.x; ++thread_coord.x)
325  {
326  ValueT running_total = 0.0;
327  for (; thread_coord.y < row_end_offsets[thread_coord.x]; ++thread_coord.y)
328  {
329  running_total += values[thread_coord.y] * vector_x[column_indices[thread_coord.y]];
330  }
331 
332  vector_y_out[thread_coord.x] = running_total;
333  }
334 
335  // Consume partial portion of thread's last row
336  ValueT running_total = 0.0;
337  for (; thread_coord.y < thread_coord_end.y; ++thread_coord.y)
338  {
339  running_total += values[thread_coord.y] * vector_x[column_indices[thread_coord.y]];
340  }
341 
342  // Save carry-outs
343  row_carry_out[tid] = thread_coord_end.x;
344  value_carry_out[tid] = running_total;
345  }
346 
347  // Carry-out fix-up (rows spanning multiple threads)
348  for (int tid = 0; tid < num_threads - 1; ++tid)
349  {
350  if (row_carry_out[tid] < a.num_rows)
351  vector_y_out[row_carry_out[tid]] += value_carry_out[tid];
352  }
353 }

359 template <
360  typename ValueT,
361  typename OffsetT>
362 float TestOmpMergeCsrmv(
363  CsrMatrix<ValueT, OffsetT>& a,
364  ValueT* vector_x,
365  ValueT* reference_vector_y_out,
366  ValueT* vector_y_out,
367  int timing_iterations,
368  float &setup_ms)
369 {
370  setup_ms = 0.0;
371 
372  if (g_omp_threads == -1)
373  g_omp_threads = omp_get_num_procs();
374  int num_threads = g_omp_threads;
375 
376  if (!g_quiet)
377  printf("\tUsing %d threads on %d procs\n", g_omp_threads, omp_get_num_procs());
378 
379  // Warmup/correctness
380  memset(vector_y_out, -1, sizeof(ValueT) * a.num_rows);
381  OmpMergeCsrmv(g_omp_threads, a, a.row_offsets + 1, a.column_indices, a.values, vector_x, vector_y_out);
382  if (!g_quiet)
383  {
384  // Check answer
385  int compare = CompareResults(reference_vector_y_out, vector_y_out, a.num_rows, true);
386  printf("\t%s\n", compare ? "FAIL" : "PASS"); fflush(stdout);
387  }
388 
389  // Re-populate caches, etc.
390  OmpMergeCsrmv(g_omp_threads, a, a.row_offsets + 1, a.column_indices, a.values, vector_x, vector_y_out);
391  OmpMergeCsrmv(g_omp_threads, a, a.row_offsets + 1, a.column_indices, a.values, vector_x, vector_y_out);
392  OmpMergeCsrmv(g_omp_threads, a, a.row_offsets + 1, a.column_indices, a.values, vector_x, vector_y_out);
393 
394  // Timing
395  float elapsed_ms = 0.0;
396  CpuTimer timer;
397  timer.Start();
398  for(int it = 0; it < timing_iterations; ++it)
399  {
400  OmpMergeCsrmv(g_omp_threads, a, a.row_offsets + 1, a.column_indices, a.values, vector_x, vector_y_out);
401  }
402  timer.Stop();
403  elapsed_ms += timer.ElapsedMillis();
404 
405  return elapsed_ms / timing_iterations;
406 }

501 template <typename ValueT, typename OffsetT>
502 void DisplayPerf(
503  double setup_ms,
504  double avg_ms,
505  CsrMatrix<ValueT, OffsetT>& csr_matrix)
506 {
507  double nz_throughput, effective_bandwidth;
508  size_t total_bytes = (csr_matrix.num_nonzeros * (sizeof(ValueT) * 2 + sizeof(OffsetT))) +
509  (csr_matrix.num_rows) * (sizeof(OffsetT) + sizeof(ValueT));
510 
511  nz_throughput = double(csr_matrix.num_nonzeros) / avg_ms / 1.0e6;
512  effective_bandwidth = double(total_bytes) / avg_ms / 1.0e6;
513 
514  if (!g_quiet)
515  printf("fp%d: %.4f setup ms, %.4f avg ms, %.5f gflops, %.3lf effective GB/s\n",
516  int(sizeof(ValueT) * 8),
517  setup_ms,
518  avg_ms,
519  2 * nz_throughput,
520  effective_bandwidth);
521  else
522  printf("%.5f, %.5f, %.6f, %.3lf, ",
523  setup_ms, avg_ms,
524  2 * nz_throughput,
525  effective_bandwidth);
526 
527  fflush(stdout);
528 }

534 template <
535  typename ValueT,
536  typename OffsetT>
537 void RunTests(
538  ValueT alpha,
539  ValueT beta,
540  const std::string& mtx_filename,
541  int grid2d,
542  int grid3d,
543  int wheel,
544  int dense,
545  int timing_iterations,
546  CommandLineArgs& args)
547 {
548  // Initialize matrix in COO form
549  CooMatrix<ValueT, OffsetT> coo_matrix;
550 
551  if (!mtx_filename.empty())
552  {
553  // Parse matrix market file
554  coo_matrix.InitMarket(mtx_filename, 1.0, !g_quiet);
555 
556  if ((coo_matrix.num_rows == 1) || (coo_matrix.num_cols == 1) || (coo_matrix.num_nonzeros == 1))
557  {
558  if (!g_quiet) printf("Trivial dataset\n");
559  exit(0);
560  }
561  printf("%s, ", mtx_filename.c_str()); fflush(stdout);
562  }
563  else if (grid2d > 0)
564  {
565  // Generate 2D lattice
566  printf("grid2d_%d, ", grid2d); fflush(stdout);
567  coo_matrix.InitGrid2d(grid2d, false);
568  }
569  else if (grid3d > 0)
570  {
571  // Generate 3D lattice
572  printf("grid3d_%d, ", grid3d); fflush(stdout);
573  coo_matrix.InitGrid3d(grid3d, false);
574  }
575  else if (wheel > 0)
576  {
577  // Generate wheel graph
578  printf("wheel_%d, ", grid2d); fflush(stdout);
579  coo_matrix.InitWheel(wheel);
580  }
581  else if (dense > 0)
582  {
583  // Generate dense graph
584  OffsetT rows = (1<<24) / dense; // 16M nnz
585  printf("dense_%d_x_%d, ", rows, dense); fflush(stdout);
586  coo_matrix.InitDense(rows, dense);
587  }
588  else
589  {
590  fprintf(stderr, "No graph type specified.\n");
591  exit(1);
592  }
593 
594  CsrMatrix<ValueT, OffsetT> csr_matrix(coo_matrix);
595  coo_matrix.Clear();
596 
597  // Display matrix info
598  csr_matrix.Stats().Display(!g_quiet);
599  if (!g_quiet)
600  {
601  printf("\n");
602  csr_matrix.DisplayHistogram();
603  printf("\n");
604  if (g_verbose2)
605  csr_matrix.Display();
606  printf("\n");
607  }
608  fflush(stdout);
609 
610  // Determine # of timing iterations (aim to run 16 billion nonzeros through, total)
611  if (timing_iterations == -1)
612  {
613  timing_iterations = std::min(200000ull, std::max(100ull, ((16ull << 30) / csr_matrix.num_nonzeros)));
614  if (!g_quiet)
615  printf("\t%d timing iterations\n", timing_iterations);
616  }
617 
618  // Allocate input and output vectors (if available, use NUMA allocation to force storage on the
619  // sockets for performance consistency)
620  ValueT *vector_x, *vector_y_in, *reference_vector_y_out, *vector_y_out;
621  if (csr_matrix.IsNumaMalloc())
622  {
623  vector_x = (ValueT*) numa_alloc_onnode(sizeof(ValueT) * csr_matrix.num_cols, 0);
624  vector_y_in = (ValueT*) numa_alloc_onnode(sizeof(ValueT) * csr_matrix.num_rows, 0);
625  reference_vector_y_out = (ValueT*) numa_alloc_onnode(sizeof(ValueT) * csr_matrix.num_rows, 0);
626  vector_y_out = (ValueT*) numa_alloc_onnode(sizeof(ValueT) * csr_matrix.num_rows, 0);
627  }
628  else
629  {
630  vector_x = (ValueT*) mkl_malloc(sizeof(ValueT) * csr_matrix.num_cols, 4096);
631  vector_y_in = (ValueT*) mkl_malloc(sizeof(ValueT) * csr_matrix.num_rows, 4096);
632  reference_vector_y_out = (ValueT*) mkl_malloc(sizeof(ValueT) * csr_matrix.num_rows, 4096);
633  vector_y_out = (ValueT*) mkl_malloc(sizeof(ValueT) * csr_matrix.num_rows, 4096);
634  }
635 
636  for (int col = 0; col < csr_matrix.num_cols; ++col)
637  vector_x[col] = 1.0;
638 
639  for (int row = 0; row < csr_matrix.num_rows; ++row)
640  vector_y_in[row] = 1.0;
641 
642  // Compute reference answer
643  SpmvGold(csr_matrix, vector_x, vector_y_in, reference_vector_y_out, alpha, beta);
644 
645  float avg_ms, setup_ms;
646 
647  // MKL SpMV
648  if (!g_quiet) printf("\n\n");
649  printf("MKL CsrMV, "); fflush(stdout);
650  avg_ms = TestMklCsrmv(csr_matrix, vector_x, reference_vector_y_out, vector_y_out, timing_iterations, setup_ms);
651  DisplayPerf(setup_ms, avg_ms, csr_matrix);
652 
653  // Merge SpMV
654  if (!g_quiet) printf("\n\n");
655  printf("Merge CsrMV, "); fflush(stdout);
656  avg_ms = TestOmpMergeCsrmv(csr_matrix, vector_x, reference_vector_y_out, vector_y_out, timing_iterations, setup_ms);
657  DisplayPerf(setup_ms, avg_ms, csr_matrix);
658 
659  // Cleanup
660  if (csr_matrix.IsNumaMalloc())
661  {
662  if (vector_x) numa_free(vector_x, sizeof(ValueT) * csr_matrix.num_cols);
663  if (vector_y_in) numa_free(vector_y_in, sizeof(ValueT) * csr_matrix.num_rows);
664  if (reference_vector_y_out) numa_free(reference_vector_y_out, sizeof(ValueT) * csr_matrix.num_rows);
665  if (vector_y_out) numa_free(vector_y_out, sizeof(ValueT) * csr_matrix.num_rows);
666  }
667  else
668  {
669  if (vector_x) mkl_free(vector_x);
670  if (vector_y_in) mkl_free(vector_y_in);
671  if (reference_vector_y_out) mkl_free(reference_vector_y_out);
672  if (vector_y_out) mkl_free(vector_y_out);
673  }
674 
675 }
