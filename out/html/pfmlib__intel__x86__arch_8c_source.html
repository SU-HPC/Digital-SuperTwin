53 /*
54  * create architected event table
55  */
56 static int
57 create_arch_event_table(unsigned int mask, int version)
58 {
59  intel_x86_entry_t *pe;
60  int i, num_events = 0;
61  int m;
62 
63  DPRINT("version=%d evt_msk=0x%x\n", version, mask);
64 
65  /*
66  * first pass: count the number of supported events
67  */
68  m = mask;
69  for(i=0; i < 7; i++, m>>=1) {
70  if ((m & 0x1) == 0)
71  num_events++;
72  }
73  intel_x86_arch_support.pme_count = num_events;
74 
75  pe = calloc(num_events, sizeof(intel_x86_entry_t));
76  if (pe == NULL)
77  return PFM_ERR_NOTSUPP;
78 
79  x86_arch_pe = pe;
80  intel_x86_arch_support.pe = pe;
81 
82  /*
83  * second pass: populate the table
84  */
85  m = mask;
86  for(i=0; i < 7; i++, m>>=1) {
87  if (!(m & 0x1)) {
88  *pe = intel_x86_arch_pe[i];
89 
90  switch(version) {
91  case 4:
92  pe->modmsk = INTEL_V4_ATTRS;
93  break;
94  case 3:
95  pe->modmsk = INTEL_V3_ATTRS;
96  break;
97  default:
98  pe->modmsk = INTEL_V2_ATTRS;
99  break;
100  }
101  pe++;
102  }
103  }
104  return PFM_SUCCESS;
105 }

107 static int
108 check_arch_pmu(int family)
109 {
110  union {
111  unsigned int val;
112  intel_x86_pmu_eax_t eax;
113  intel_x86_pmu_edx_t edx;
114  } eax, ecx, edx, ebx;
115 
116  /*
117  * check family number to reject for processors
118  * older than Pentium (family=5). Those processors
119  * did not have the CPUID instruction
120  */
121  if (family < 5 || family == 15)
122  return PFM_ERR_NOTSUPP;
123 
124  /*
125  * check if CPU supports 0xa function of CPUID
126  * 0xa started with Core Duo. Needed to detect if
127  * architected PMU is present
128  */
129  cpuid(0x0, &eax.val, &ebx.val, &ecx.val, &edx.val);
130  if (eax.val < 0xa)
131  return PFM_ERR_NOTSUPP;
132 
133  /*
134  * extract architected PMU information
135  */
136  cpuid(0xa, &eax.val, &ebx.val, &ecx.val, &edx.val);
137 
138  /*
139  * version must be greater than zero
140  */
141  return eax.eax.version < 1 ? PFM_ERR_NOTSUPP : PFM_SUCCESS;
142 }

156 static int
157 pfm_intel_x86_arch_init(void *this)
158 {
159  union {
160  unsigned int val;
161  intel_x86_pmu_eax_t eax;
162  intel_x86_pmu_edx_t edx;
163  } eax, ecx, edx, ebx;
164 
165  /*
166  * extract architected PMU information
167  */
168  if (!pfm_cfg.forced_pmu) {
169  cpuid(0xa, &eax.val, &ebx.val, &ecx.val, &edx.val);
170  intel_x86_arch_support.num_cntrs = eax.eax.num_cnt;
171  intel_x86_arch_support.num_fixed_cntrs = edx.edx.num_cnt;
172  } else {
173  eax.eax.version = 3;
174  ebx.val = 0; /* no restriction */
175  intel_x86_arch_support.num_cntrs = 0;
176  intel_x86_arch_support.num_fixed_cntrs = 0;
177  }
178  /*
179  * must be called after impl_cntrs has been initialized
180  */
181  return create_arch_event_table(ebx.val, eax.eax.version);
182 }
