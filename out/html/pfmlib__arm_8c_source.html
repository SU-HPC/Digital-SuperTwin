60 static int
61 pfmlib_getcpuinfo_attr(const char *attr, char *ret_buf, size_t maxlen)
62 {
63  FILE *fp = NULL;
64  int ret = -1;
65  size_t attr_len, buf_len = 0;
66  char *p, *value = NULL;
67  char *buffer = NULL;
68 
69  if (attr == NULL || ret_buf == NULL || maxlen < 1)
70  return -1;
71 
72  attr_len = strlen(attr);
73 
74  fp = fopen("/proc/cpuinfo", "r");
75  if (fp == NULL)
76  return -1;
77 
78  while(pfmlib_getl(&buffer, &buf_len, fp) != -1){
79 
80  /* skip blank lines */
81  if (*buffer == '\n')
82  continue;
83 
84  p = strchr(buffer, ':');
85  if (p == NULL)
86  goto error;
87 
88  /*
89  * p+2: +1 = space, +2= firt character
90  * strlen()-1 gets rid of \n
91  */
92  *p = '\0';
93  value = p+2;
94 
95  value[strlen(value)-1] = '\0';
96 
97  if (!strncmp(attr, buffer, attr_len))
98  break;
99  }
100  strncpy(ret_buf, value, maxlen-1);
101  ret_buf[maxlen-1] = '\0';
102  ret = 0;
103 error:
104  free(buffer);
105  fclose(fp);
106  return ret;
107 }

178 int
179 pfm_arm_get_encoding(void *this, pfmlib_event_desc_t *e)
180 {
181 
182  const arm_entry_t *pe = this_pe(this);
183  pfmlib_event_attr_info_t *a;
184  pfm_arm_reg_t reg;
185  unsigned int plm = 0;
186  int i, idx, has_plm = 0;
187 
188  reg.val = pe[e->event].code;
189 
190 
191  for (i = 0; i < e->nattrs; i++) {
192  a = attr(e, i);
193 
194  if (a->ctrl != PFM_ATTR_CTRL_PMU)
195  continue;
196 
197  if (a->type > PFM_ATTR_UMASK) {
198  uint64_t ival = e->attrs[i].ival;
199 
200  switch(a->idx) {
201  case ARM_ATTR_U: /* USR */
202  if (ival)
203  plm |= PFM_PLM3;
204  has_plm = 1;
205  break;
206  case ARM_ATTR_K: /* OS */
207  if (ival)
208  plm |= PFM_PLM0;
209  has_plm = 1;
210  break;
211  case ARM_ATTR_HV: /* HYPERVISOR */
212  if (ival)
213  plm |= PFM_PLMH;
214  has_plm = 1;
215  break;
216  default:
217  return PFM_ERR_ATTR;
218  }
219  }
220  }
221 
222  if (arm_has_plm(this, e)) {
223  if (!has_plm)
224  plm = e->dfl_plm;
225  reg.evtsel.excl_pl1 = !(plm & PFM_PLM0);
226  reg.evtsel.excl_usr = !(plm & PFM_PLM3);
227  reg.evtsel.excl_hyp = !(plm & PFM_PLMH);
228  }
229 
230  evt_strcat(e->fstr, "%s", pe[e->event].name);
231 
232  e->codes[0] = reg.val;
233  e->count = 1;
234 
235  for (i = 0; i < e->npattrs; i++) {
236  if (e->pattrs[i].ctrl != PFM_ATTR_CTRL_PMU)
237  continue;
238 
239  if (e->pattrs[i].type == PFM_ATTR_UMASK)
240  continue;
241 
242  idx = e->pattrs[i].idx;
243  switch(idx) {
244  case ARM_ATTR_K:
245  evt_strcat(e->fstr, ":%s=%lu", arm_mods[idx].name, !reg.evtsel.excl_pl1);
246  break;
247  case ARM_ATTR_U:
248  evt_strcat(e->fstr, ":%s=%lu", arm_mods[idx].name, !reg.evtsel.excl_usr);
249  break;
250  case ARM_ATTR_HV:
251  evt_strcat(e->fstr, ":%s=%lu", arm_mods[idx].name, !reg.evtsel.excl_hyp);
252  break;
253  }
254  }
255 
256  pfm_arm_display_reg(this, e, reg);
257 
258  return PFM_SUCCESS;
259 }

286 pfm_arm_validate_table(void *this, FILE *fp)
287 {
288 
289  pfmlib_pmu_t *pmu = this;
290  const arm_entry_t *pe = this_pe(this);
291  int i, j, error = 0;
292 
293  for(i=0; i < pmu->pme_count; i++) {
294  if (!pe[i].name) {
295  fprintf(fp, "pmu: %s event%d: :: no name (prev event was %s)\n", pmu->name, i,
296  i > 1 ? pe[i-1].name : "??");
297  error++;
298  }
299  if (!pe[i].desc) {
300  fprintf(fp, "pmu: %s event%d: %s :: no description\n", pmu->name, i, pe[i].name);
301  error++;
302  }
303  for(j = i+1; j < pmu->pme_count; j++) {
304  if (pe[i].code == pe[j].code && !(pe[j].equiv || pe[i].equiv)) {
305  fprintf(fp, "pmu: %s events %s and %s have the same code 0x%x\n", pmu->name, pe[i].name, pe[j].name, pe[i].code);
306  error++;
307  }
308  }
309  }
310  return error ? PFM_ERR_INVAL : PFM_SUCCESS;
311 }
