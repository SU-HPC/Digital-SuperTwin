141 static int
142 match_ufilters(pfm_event_attr_info_t *info)
143 {
144  uint32_t ufilter1 = 0;

146 
147  if (options.ufilter.is_dfl)

152

191 static void
192 show_event_info_combo(pfm_event_info_t *info)
193 {
194  pfm_event_attr_info_t *ainfo;
195  pfm_pmu_info_t pinfo;
196  char buf[MAXBUF];
197  size_t len;
198  int numasks = 0;
199  int i, j, ret;
200  uint64_t total, m, u;
201 
202  memset(&pinfo, 0, sizeof(pinfo));
203 
204  pinfo.size = sizeof(pinfo);
205 
206  ret = pfm_get_pmu_info(info->pmu, &pinfo);
207  if (ret != PFM_SUCCESS)
208  errx(1, "cannot get PMU info");
209 
210  ainfo = calloc(info->nattrs, sizeof(*ainfo));
211  if (!ainfo)
212  err(1, "event %s : ", info->name);
213 
214  /*
215  * extract attribute information and count number
216  * of umasks
217  *
218  * we cannot just drop non umasks because we need
219  * to keep attributes in order for the enumeration
220  * of 2^n
221  */
222  pfm_for_each_event_attr(i, info) {
223  ainfo[i].size = sizeof(*ainfo);
224 
225  ret = pfm_get_event_attr_info(info->idx, i, options.os, &ainfo[i]);
226  if (ret != PFM_SUCCESS)
227  errx(1, "cannot get attribute info: %s", pfm_strerror(ret));
228 
229  if (ainfo[i].type == PFM_ATTR_UMASK)
230  numasks++;
231  }
232  if (numasks > options.combo_lim) {
233  warnx("event %s has too many umasks to print all combinations, dropping to simple enumeration", info->name);
234  free(ainfo);
235  show_event_info_compact(info);
236  return;
237  }
238 
239  if (numasks) {
240  if (info->nattrs > (int)((sizeof(total)<<3))) {
241  warnx("too many umasks, cannot show all combinations for event %s", info->name);
242  goto end;
243  }
244  total = 1ULL << info->nattrs;
245 
246  for (u = 1; u < total; u++) {
247  len = sizeof(buf);
248  len -= snprintf(buf, len, "%s::%s", pinfo.name, info->name);
249  if (len <= 0) {
250  warnx("event name too long%s", info->name);
251  goto end;
252  }
253  for(m = u, j = 0; m; m >>=1, j++) {
254  if (m & 0x1ULL) {
255  /* we have hit a non umasks attribute, skip */
256  if (ainfo[j].type != PFM_ATTR_UMASK)
257  break;
258 
259  if (len < (1 + strlen(ainfo[j].name))) {
260  warnx("umasks combination too long for event %s", buf);
261  break;
262  }
263  strncat(buf, ":", len-1);buf[len-1] = '\0'; len--;
264  strncat(buf, ainfo[j].name, len-1);buf[len-1] = '\0';
265  len -= strlen(ainfo[j].name);
266  }
267  }
268  /* if found a valid umask combination, check encoding */
269  if (m == 0) {
270  if (options.encode)
271  ret = print_codes(buf, PFM_PLM0|PFM_PLM3, pinfo.max_encoding);
272  else
273  ret = check_valid(buf, PFM_PLM0|PFM_PLM3);
274  if (!ret)
275  printf("%s\n", buf);
276  }
277  }
278  } else {
279  snprintf(buf, sizeof(buf)-1, "%s::%s", pinfo.name, info->name);
280  buf[sizeof(buf)-1] = '\0';
281 
282  ret = options.encode ? print_codes(buf, PFM_PLM0|PFM_PLM3, pinfo.max_encoding) : 0;
283  if (!ret)
284  printf("%s\n", buf);
285  }
286 end:
287  free(ainfo);
288 }

570  if (options.compact)

574  show_event_info_compact(&info);

578  }
