59 static int
60 pfmlib_getcpuinfo_attr(const char *attr, char *ret_buf, size_t maxlen)
61 {
62  FILE *fp = NULL;
63  int ret = -1;
64  size_t attr_len, buf_len = 0;
65  char *p, *value = NULL;
66  char *buffer = NULL;
67 
68  if (attr == NULL || ret_buf == NULL || maxlen < 1)
69  return -1;
70 
71  attr_len = strlen(attr);
72 
73  fp = fopen("/proc/cpuinfo", "r");
74  if (fp == NULL)
75  return -1;
76 
77  while(pfmlib_getl(&buffer, &buf_len, fp) != -1){
78 
79  /* skip blank lines */
80  if (*buffer == '\n')
81  continue;
82 
83  p = strchr(buffer, ':');
84  if (p == NULL)
85  goto error;
86 
87  /*
88  * p+2: +1 = space, +2= firt character
89  * strlen()-1 gets rid of \n
90  */
91  *p = '\0';
92  value = p+2;
93 
94  value[strlen(value)-1] = '\0';
95 
96  if (!strncmp(attr, buffer, attr_len))
97  break;
98  }
99  strncpy(ret_buf, value, maxlen-1);
100  ret_buf[maxlen-1] = '\0';
101  ret = 0;
102 error:
103  free(buffer);
104  fclose(fp);
105  return ret;
106 }

237 
238  pfmlib_pmu_t *pmu = this;
239  const sparc_entry_t *pe = this_pe(this);

241 
242  for(i=0; i < pmu->pme_count; i++) {
243  if (!pe[i].name) {
244  fprintf(fp, "pmu: %s event%d: :: no name (prev event was %s)\n", pmu->name, i,
245  i > 1 ? pe[i-1].name : "??");
246  error++;
247  }
248  if (!pe[i].desc) {
249  fprintf(fp, "pmu: %s event%d: %s :: no description\n", pmu->name, i, pe[i].name);
250  error++;
251  }
252  for(j=i+1; j < pmu->pme_count; j++) {
253  if (pe[i].code == pe[j].code && pe[i].ctrl == pe[j].ctrl) {
254  fprintf(fp, "pmu: %s event%d: %s code: 0x%x is duplicated in event%d : %s\n", pmu->name, i, pe[i].name, pe[i].code, j, pe[j].name);
255  error++;
256  }
257  }
258  }
259  return error ? PFM_ERR_INVAL : PFM_SUCCESS;
260 }
