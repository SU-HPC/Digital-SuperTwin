62 /* Parses the machine type that identifies an IBM mainframe.
63  * These kind of information are from /proc/sysinfo.
64  */
65 static long get_machine_type(void)
66 {
67  long machine_type;
68  size_t buflen, len;
69  char *buffer, *tmp;
70  FILE *fp;
71 
72  machine_type = 0;
73  fp = fopen(SYS_INFO, "r");
74  if (fp == NULL)
75  goto out;
76 
77  buffer = NULL;
78  while (pfmlib_getl(&buffer, &buflen, fp) != -1) {
79  /* skip empty lines */
80  if (*buffer == '\n')
81  continue;
82 
83  /* look for 'Type:' entry */
84  if (!strncmp("Type:", buffer, 5)) {
85  tmp = buffer + 5; /* set ptr after ':' */
86  /* skip leading blanks */
87  while (isspace(*tmp))
88  tmp++;
89  /* skip trailing blanks */
90  len = strlen(tmp);
91  while (len > 0 && isspace(tmp[len]))
92  len--;
93  tmp[len+1] = '\0';
94  machine_type = strtol(tmp, NULL, 10);
95  break;
96  }
97  }
98  fclose(fp);
99  free(buffer);
100 out:
101  return machine_type;
102 }

104 static void get_cf_version(unsigned int *cfvn, unsigned int *csvn)
105 {
106  int rc;
107  FILE *fp;
108  char *buffer;
109  size_t buflen;
110 
111  *cfvn = *csvn = 0;
112  fp = fopen(SERVICE_LEVEL, "r");
113  if (fp == NULL)
114  return;
115 
116  buffer = NULL;
117  while (pfmlib_getl(&buffer, &buflen, fp) != -1) {
118  /* skip empty lines */
119  if (*buffer == '\n')
120  continue;
121 
122  /* look for 'CPU-MF: Counter facility: version=' entry */
123  if (!strncmp(CF_VERSION_STR, buffer, strlen(CF_VERSION_STR))) {
124  rc = sscanf(buffer + strlen(CF_VERSION_STR), "%u.%u",
125  cfvn, csvn);
126  if (rc != 2)
127  *cfvn = *csvn = 0;
128  break;
129  }
130  }
131  fclose(fp);
132  free(buffer);
133 }

135 /* Initialize the PMU representation for CPUMF.
136  *
137  * Set up the PMU events array based on
138  * - generic (basic, problem-state, and crypto-activaty) counter sets
139  * - the extended counter depending on the machine type
140  */
141 static int pfm_cpumcf_init(void *this)
142 {
143  pfmlib_pmu_t *pmu = this;
144  unsigned int cfvn, csvn;
145  const pme_cpumf_ctr_t *cfvn_set, *csvn_set, *ext_set;
146  size_t cfvn_set_count, csvn_set_count, ext_set_count, pme_count;
147 
148  /* obtain counter first/second version number */
149  get_cf_version(&cfvn, &csvn);
150 
151  /* counters based on first version number */
152  switch (cfvn)
153  {
154  case 1:
155  cfvn_set = cpumcf_fvn1_counters;
156  cfvn_set_count = LIBPFM_ARRAY_SIZE(cpumcf_fvn1_counters);
157  break;
158  case 3:
159  cfvn_set = cpumcf_fvn3_counters;
160  cfvn_set_count = LIBPFM_ARRAY_SIZE(cpumcf_fvn3_counters);
161  break;
162  default:
163  cfvn_set = NULL;
164  cfvn_set_count = 0;
165  break;
166  }
167 
168  /* counters based on second version number */
169  csvn_set = cpumcf_svn_generic_counters;
170  csvn_set_count = LIBPFM_ARRAY_SIZE(cpumcf_svn_generic_counters);
171  if (csvn < 6) /* Crypto counter set enlarged for SVN == 6 */
172  csvn_set_count -= CPUMF_SVN6_ECC;
173 
174  /* check and assign a machine-specific extended counter set */
175  switch (get_machine_type()) {
176  case 2097: /* IBM System z10 EC */
177  case 2098: /* IBM System z10 BC */
178  ext_set = cpumcf_z10_counters,
179  ext_set_count = LIBPFM_ARRAY_SIZE(cpumcf_z10_counters);
180  break;
181  case 2817: /* IBM zEnterprise 196 */
182  case 2818: /* IBM zEnterprise 114 */
183  ext_set = cpumcf_z196_counters;
184  ext_set_count = LIBPFM_ARRAY_SIZE(cpumcf_z196_counters);
185  break;
186  case 2827: /* IBM zEnterprise EC12 */
187  case 2828: /* IBM zEnterprise BC12 */
188  ext_set = cpumcf_zec12_counters;
189  ext_set_count = LIBPFM_ARRAY_SIZE(cpumcf_zec12_counters);
190  break;
191  case 2964: /* IBM z13 */
192  case 2965: /* IBM z13s */
193  ext_set = cpumcf_z13_counters;
194  ext_set_count = LIBPFM_ARRAY_SIZE(cpumcf_z13_counters);
195  break;
196  case 3906: /* IBM z14 */
197  case 3907: /* IBM z14 ZR1 */
198  ext_set = cpumcf_z14_counters;
199  ext_set_count = LIBPFM_ARRAY_SIZE(cpumcf_z14_counters);
200  break;
201  case 8561: /* IBM Machine types 8561 and 8562 */
202  case 8562:
203  ext_set = cpumcf_z15_counters;
204  ext_set_count = LIBPFM_ARRAY_SIZE(cpumcf_z15_counters);
205  break;
206  default:
207  /* No extended counter set for this machine type or there
208  * was an error retrieving the machine type */
209  ext_set = NULL;
210  ext_set_count = 0;
211  break;
212  }
213 
214  cpumcf_pe = calloc(sizeof(*cpumcf_pe),
215  cfvn_set_count + csvn_set_count + ext_set_count);
216  if (cpumcf_pe == NULL)
217  return PFM_ERR_NOMEM;
218 
219  pme_count = 0;
220  memcpy(cpumcf_pe, cfvn_set, sizeof(*cpumcf_pe) * cfvn_set_count);
221  pme_count += cfvn_set_count;
222  memcpy((void *) (cpumcf_pe + pme_count), csvn_set,
223  sizeof(*cpumcf_pe) * csvn_set_count);
224  pme_count += csvn_set_count;
225  if (ext_set_count)
226  memcpy((void *) (cpumcf_pe + pme_count),
227  ext_set, sizeof(*cpumcf_pe) * ext_set_count);
228  pme_count += ext_set_count;
229 
230  pmu->pe = cpumcf_pe;
231  pmu->pme_count = pme_count;
232 
233  /* CPUM-CF provides fixed counters only. The number of installed
234  * counters depends on the version and hardware model up to
235  * CPUMF_COUNTER_MAX.
236  */
237  pmu->num_fixed_cntrs = pme_count;
238 
239  return PFM_SUCCESS;
240 }
