44 static int
45 pfm_arm_detect_n1(void *this)
46 {
47  int ret;
48 
49  ret = pfm_arm_detect(this);
50  if (ret != PFM_SUCCESS)
51  return PFM_ERR_NOTSUPP;
52 
53  if ((pfm_arm_cfg.implementer == 0x41) && /* ARM */
54  (pfm_arm_cfg.part == 0xd0c)) { /* Neoverse N1 */
55  return PFM_SUCCESS;
56  }
57  return PFM_ERR_NOTSUPP;
58 }
