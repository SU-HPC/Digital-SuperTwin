77 const unsigned int N = 624;
78 const unsigned int M = 397;
79 const unsigned int MATRIX_A = 0x9908b0df; /* constant vector a */
80 const unsigned int UPPER_MASK = 0x80000000; /* most significant w-r bits */
81 const unsigned int LOWER_MASK = 0x7fffffff; /* least significant r bits */

83 static unsigned int mt[N]; /* the array for the state vector */
84 static int mti = N + 1; /* mti==N+1 means mt[N] is not initialized */

86 /* initializes mt[N] with a seed */
87 void init_genrand(unsigned int s)
88 {
89  mt[0] = s & 0xffffffff;
90  for (mti = 1; mti < N; mti++)
91  {
92  mt[mti] = (1812433253 * (mt[mti - 1] ^ (mt[mti - 1] >> 30)) + mti);
93 
94  /* See Knuth TAOCP Vol2. 3rd Ed. P.106 for mtiplier. */
95  /* In the previous versions, MSBs of the seed affect */
96  /* only MSBs of the array mt[]. */
97  /* 2002/01/09 modified by Makoto Matsumoto */
98 
99  mt[mti] &= 0xffffffff;
100  /* for >32 bit machines */
101  }
102 }

104 /* initialize by an array with array-length */
105 /* init_key is the array for initializing keys */
106 /* key_length is its length */
107 /* slight change for C++, 2004/2/26 */
108 void init_by_array(unsigned int init_key[], int key_length)
109 {
110  int i, j, k;
111  init_genrand(19650218);
112  i = 1;
113  j = 0;
114  k = (N > key_length ? N : key_length);
115  for (; k; k--)
116  {
117  mt[i] = (mt[i] ^ ((mt[i - 1] ^ (mt[i - 1] >> 30)) * 1664525))
118  + init_key[j] + j; /* non linear */
119  mt[i] &= 0xffffffff; /* for WORDSIZE > 32 machines */
120  i++;
121  j++;
122  if (i >= N)
123  {
124  mt[0] = mt[N - 1];
125  i = 1;
126  }
127  if (j >= key_length) j = 0;
128  }
129  for (k = N - 1; k; k--)
130  {
131  mt[i] = (mt[i] ^ ((mt[i - 1] ^ (mt[i - 1] >> 30)) * 1566083941)) - i; /* non linear */
132  mt[i] &= 0xffffffff; /* for WORDSIZE > 32 machines */
133  i++;
134  if (i >= N)
135  {
136  mt[0] = mt[N - 1];
137  i = 1;
138  }
139  }
140 
141  mt[0] = 0x80000000; /* MSB is 1; assuring non-zero initial array */
142 }

144 /* generates a random number on [0,0xffffffff]-interval */
145 unsigned int genrand_int32(void)
146 {
147  unsigned int y;
148  static unsigned int mag01[2] = { 0x0, MATRIX_A };
149 
150  /* mag01[x] = x * MATRIX_A for x=0,1 */
151 
152  if (mti >= N)
153  { /* generate N words at one time */
154  int kk;
155 
156  if (mti == N + 1) /* if init_genrand() has not been called, */
157  init_genrand(5489); /* a defat initial seed is used */
158 
159  for (kk = 0; kk < N - M; kk++)
160  {
161  y = (mt[kk] & UPPER_MASK) | (mt[kk + 1] & LOWER_MASK);
162  mt[kk] = mt[kk + M] ^ (y >> 1) ^ mag01[y & 0x1];
163  }
164  for (; kk < N - 1; kk++)
165  {
166  y = (mt[kk] & UPPER_MASK) | (mt[kk + 1] & LOWER_MASK);
167  mt[kk] = mt[kk + (M - N)] ^ (y >> 1) ^ mag01[y & 0x1];
168  }
169  y = (mt[N - 1] & UPPER_MASK) | (mt[0] & LOWER_MASK);
170  mt[N - 1] = mt[M - 1] ^ (y >> 1) ^ mag01[y & 0x1];
171 
172  mti = 0;
173  }
174 
175  y = mt[mti++];
176 
177  /* Tempering */
178  y ^= (y >> 11);
179  y ^= (y << 7) & 0x9d2c5680;
180  y ^= (y << 15) & 0xefc60000;
181  y ^= (y >> 18);
182 
183  return y;
184 }

213 template <typename K>
214 void RandomBits(
215  K &key,
216  int entropy_reduction = 0,
217  int begin_bit = 0,
218  int end_bit = sizeof(K) * 8)
219 {
220  const int NUM_BYTES = sizeof(K);
221  const int WORD_BYTES = sizeof(unsigned int);
222  const int NUM_WORDS = (NUM_BYTES + WORD_BYTES - 1) / WORD_BYTES;
223 
224  unsigned int word_buff[NUM_WORDS];
225 
226  if (entropy_reduction == -1)
227  {
228  memset((void *) &key, 0, sizeof(key));
229  return;
230  }
231 
232  if (end_bit < 0)
233  end_bit = sizeof(K) * 8;
234 
235  // Generate random word_buff
236  for (int j = 0; j < NUM_WORDS; j++)
237  {
238  int current_bit = j * WORD_BYTES * 8;
239 
240  unsigned int word = 0xffffffff;
241  word &= 0xffffffff << std::max(0, begin_bit - current_bit);
242  word &= 0xffffffff >> std::max(0, (current_bit + (WORD_BYTES * 8)) - end_bit);
243 
244  for (int i = 0; i <= entropy_reduction; i++)
245  {
246  // Grab some of the higher bits from rand (better entropy, supposedly)
247  word &= mersenne::genrand_int32();
248  g_num_rand_samples++;
249  }
250 
251  word_buff[j] = word;
252  }
253 
254  memcpy(&key, word_buff, sizeof(K));
255 }
