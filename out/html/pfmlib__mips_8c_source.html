61 static int
62 pfmlib_getcpuinfo_attr(const char *attr, char *ret_buf, size_t maxlen)
63 {
64  FILE *fp = NULL;
65  int ret = -1;
66  size_t attr_len, buf_len = 0;
67  char *p, *value = NULL;
68  char *buffer = NULL;
69 
70  if (attr == NULL || ret_buf == NULL || maxlen < 1)
71  return -1;
72 
73  attr_len = strlen(attr);
74 
75  fp = fopen("/proc/cpuinfo", "r");
76  if (fp == NULL)
77  return -1;
78 
79  while(pfmlib_getl(&buffer, &buf_len, fp) != -1){
80 
81  /* skip blank lines */
82  if (*buffer == '\n')
83  continue;
84 
85  p = strchr(buffer, ':');
86  if (p == NULL)
87  goto error;
88 
89  /*
90  * p+2: +1 = space, +2= firt character
91  * strlen()-1 gets rid of \n
92  */
93  *p = '\0';
94  value = p+2;
95 
96  value[strlen(value)-1] = '\0';
97 
98  if (!strncmp(attr, buffer, attr_len))
99  break;
100  }
101  strncpy(ret_buf, value, maxlen-1);
102  ret_buf[maxlen-1] = '\0';
103  ret = 0;
104 error:
105  free(buffer);
106  fclose(fp);
107  return ret;
108 }

171 int
172 pfm_mips_get_encoding(void *this, pfmlib_event_desc_t *e)
173 {
174 
175  pfmlib_pmu_t *pmu = this;
176  const mips_entry_t *pe = this_pe(this);
177  pfmlib_event_attr_info_t *a;
178  pfm_mips_sel_reg_t reg;
179  uint64_t ival, cntmask = 0;
180  int plmmsk = 0, code;
181  int k, id;
182 
183  reg.val = 0;
184  code = pe[e->event].code;
185 
186  /* truncates bit 7 (counter info) */
187  reg.perfsel64.sel_event_mask = code;
188 
189  for (k = 0; k < e->nattrs; k++) {
190  a = attr(e, k);
191 
192  if (a->ctrl != PFM_ATTR_CTRL_PMU)
193  continue;
194 
195  ival = e->attrs[k].ival;
196  switch(a->idx) {
197  case MIPS_ATTR_K: /* os */
198  reg.perfsel64.sel_os = !!ival;
199  plmmsk |= _MIPS_ATTR_K;
200  break;
201  case MIPS_ATTR_U: /* user */
202  reg.perfsel64.sel_usr = !!ival;
203  plmmsk |= _MIPS_ATTR_U;
204  break;
205  case MIPS_ATTR_S: /* supervisor */
206  reg.perfsel64.sel_sup = !!ival;
207  plmmsk |= _MIPS_ATTR_S;
208  break;
209  case MIPS_ATTR_E: /* int */
210  reg.perfsel64.sel_exl = !!ival;
211  plmmsk |= _MIPS_ATTR_E;
212  }
213  }
214 
215  /*
216  * handle case where no priv level mask was passed.
217  * then we use the dfl_plm
218  */
219  if (!(plmmsk & MIPS_PLM_ALL)) {
220  if (e->dfl_plm & PFM_PLM0)
221  reg.perfsel64.sel_os = 1;
222  if (e->dfl_plm & PFM_PLM1)
223  reg.perfsel64.sel_sup = 1;
224  if (e->dfl_plm & PFM_PLM2)
225  reg.perfsel64.sel_exl = 1;
226  if (e->dfl_plm & PFM_PLM3)
227  reg.perfsel64.sel_usr = 1;
228  }
229 
230  evt_strcat(e->fstr, "%s", pe[e->event].name);
231 
232  for (k = 0; k < e->npattrs; k++) {
233 
234  if (e->pattrs[k].ctrl != PFM_ATTR_CTRL_PMU)
235  continue;
236 
237  id = e->pattrs[k].idx;
238  switch(id) {
239  case MIPS_ATTR_K:
240  evt_strcat(e->fstr, ":%s=%lu", mips_mods[id].name, reg.perfsel64.sel_os);
241  break;
242  case MIPS_ATTR_U:
243  evt_strcat(e->fstr, ":%s=%lu", mips_mods[id].name, reg.perfsel64.sel_usr);
244  break;
245  case MIPS_ATTR_S:
246  evt_strcat(e->fstr, ":%s=%lu", mips_mods[id].name, reg.perfsel64.sel_sup);
247  break;
248  case MIPS_ATTR_E:
249  evt_strcat(e->fstr, ":%s=%lu", mips_mods[id].name, reg.perfsel64.sel_exl);
250  break;
251  }
252  }
253  e->codes[0] = reg.val;
254 
255  /* cycles and instructions support all counters */
256  if (code == 0 || code == 1) {
257  cntmask = (1ULL << pmu->num_cntrs) -1;
258  } else {
259  /* event work on odd counters only */
260  for (k = !!(code & 0x80) ; k < pmu->num_cntrs; k+=2) {
261  cntmask |= 1ULL << k;
262  }
263  }
264  e->codes[1] = cntmask;
265  e->count = 2;
266 
267  pfm_mips_display_reg(reg, cntmask, e->fstr);
268 
269  return PFM_SUCCESS;
270 }

297 pfm_mips_validate_table(void *this, FILE *fp)
298 {
299 
300  pfmlib_pmu_t *pmu = this;
301  const mips_entry_t *pe = this_pe(this);
302  int i, j, error = 0;
303 
304  for(i=0; i < pmu->pme_count; i++) {
305  if (!pe[i].name) {
306  fprintf(fp, "pmu: %s event%d: :: no name (prev event was %s)\n", pmu->name, i,
307  i > 1 ? pe[i-1].name : "??");
308  error++;
309  }
310  if (!pe[i].desc) {
311  fprintf(fp, "pmu: %s event%d: %s :: no description\n", pmu->name, i, pe[i].name);
312  error++;
313  }
314  for (j=i+1; j < pmu->pme_count; j++) {
315  if (pe[i].code == pe[j].code) {
316  fprintf(fp, "pmu: %s events %s and %s have the same code 0x%x\n", pmu->name, pe[i].name, pe[j].name, pe[i].code);
317  error++;
318  }
319  }
320  }
321  if (!pmu->supported_plm) {
322  fprintf(fp, "pmu: %s supported_plm=0, is that right?\n", pmu->name);
323  error++;
324  }
325  return error ? PFM_ERR_INVAL : PFM_SUCCESS;
326 }
