37 display_cbo(void *this, pfmlib_event_desc_t *e, void *val)
38 {
39  const intel_x86_entry_t *pe = this_pe(this);
40  pfm_snbep_unc_reg_t *reg = val;
41  pfm_snbep_unc_reg_t f;
42 
43  __pfm_vbprintf("[UNC_CBO=0x%"PRIx64" event=0x%x umask=0x%x en=%d "
44  "inv=%d edge=%d thres=%d tid_en=%d] %s\n",
45  reg->val,
46  reg->cbo.unc_event,
47  reg->cbo.unc_umask,
48  reg->cbo.unc_en,
49  reg->cbo.unc_inv,
50  reg->cbo.unc_edge,
51  reg->cbo.unc_thres,
52  reg->cbo.unc_tid,
53  pe[e->event].name);
54 
55  if (e->count == 1)
56  return;
57 
58  f.val = e->codes[1];
59 
60  __pfm_vbprintf("[UNC_CBOX_FILTER0=0x%"PRIx64" tid=%d core=0x%x"
61  " state=0x%x]\n",
62  f.val,
63  f.ivbep_cbo_filt0.tid,
64  f.ivbep_cbo_filt0.cid,
65  f.ivbep_cbo_filt0.state);
66 
67  if (e->count == 2)
68  return;
69 
70  f.val = e->codes[2];
71 
72  __pfm_vbprintf("[UNC_CBOX_FILTER1=0x%"PRIx64" nid=%d opc=0x%x"
73  " nc=0x%x isoc=0x%x]\n",
74  f.val,
75  f.ivbep_cbo_filt1.nid,
76  f.ivbep_cbo_filt1.opc,
