66  pfm_snbep_unc_reg_t reg;
67  int ret;
68 
69  if (!pmu->get_event_encoding[PFM_OS_NONE])
70  return PFM_ERR_NOTSUPP;
71 
72  ret = pmu->get_event_encoding[PFM_OS_NONE](this, e);
73  if (ret != PFM_SUCCESS)
74  return ret;
75 
76  ret = find_pmu_type_by_name(pmu->perf_name);
77  if (ret < 0)
78  return ret;
79 
80  attr->type = ret;
81 
82  reg.val = e->codes[0];
83 
84  attr->config = reg.val;
85 
86  if (( is_cbo_filt_event(this, reg)
87  || is_cha_filt_event(this, 0, reg)
88  || is_cha_filt_event(this, 1, reg))
89  && e->count > 1) {
90  if (e->count >= 2)
91  attr->config1 = e->codes[1];
92  if (e->count >= 3)
93  attr->config1 |= e->codes[2] << 32;
94  } else {
95  /*
96  * various filters
97  */
98  if (e->count >= 2)
99  attr->config1 = e->codes[1];
100 
101  if (e->count >= 3)
102  attr->config2 = e->codes[2];
103  }
104 
105  /*
106  * uncore measures at all priv levels
107  *
108  * user cannot set per-event priv levels because
109  * attributes are simply not there
110  *
111  * dfl_plm is ignored in this case
112  */
113  attr->exclude_hv = 0;
114  attr->exclude_kernel = 0;
115  attr->exclude_user = 0;
116 
117  return PFM_SUCCESS;

120 void
121 pfm_intel_snbep_unc_perf_validate_pattrs(void *this, pfmlib_event_desc_t *e)
122 {
123  pfmlib_pmu_t *pmu = this;
124  int no_smpl = pmu->flags & PFMLIB_PMU_FL_NO_SMPL;
125  int i, compact;
126 
127  for (i = 0; i < e->npattrs; i++) {
128  compact = 0;
129  /* umasks never conflict */
130  if (e->pattrs[i].type == PFM_ATTR_UMASK)
131  continue;
132 
133  if (e->pattrs[i].ctrl == PFM_ATTR_CTRL_PERF_EVENT) {
134 
135  /* No precise sampling mode for uncore */
136  if (e->pattrs[i].idx == PERF_ATTR_PR)
137  compact = 1;
138 
139  /*
140  * No hypervisor for uncore
141  */
142  if (e->pattrs[i].idx == PERF_ATTR_H)
143  compact = 1;
144 
145  if (no_smpl
146  && ( e->pattrs[i].idx == PERF_ATTR_FR
147  || e->pattrs[i].idx == PERF_ATTR_PR
148  || e->pattrs[i].idx == PERF_ATTR_PE))
149  compact = 1;
150 
151  /*
152  * uncore has no priv level support
153  */
154  if (pmu->supported_plm == 0
155  && ( e->pattrs[i].idx == PERF_ATTR_U
156  || e->pattrs[i].idx == PERF_ATTR_K
157  || e->pattrs[i].idx == PERF_ATTR_MG
158  || e->pattrs[i].idx == PERF_ATTR_MH))
159  compact = 1;
160  }
161 
162  /* hardware sampling not supported */
163  if (e->pattrs[i].idx == PERF_ATTR_HWS)
164  compact = 1;
165 
166  if (compact) {
167  pfmlib_compact_pattrs(e, i);
168  i--;
169  }
170  }
