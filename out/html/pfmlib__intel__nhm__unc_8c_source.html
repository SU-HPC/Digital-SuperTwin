81 static int
82 pfm_nhm_unc_get_encoding(void *this, pfmlib_event_desc_t *e)
83 {
84  pfm_intel_x86_reg_t reg;
85  pfmlib_event_attr_info_t *a;
86  const intel_x86_entry_t *pe = this_pe(this);
87  unsigned int grpmsk, ugrpmsk = 0;
88  int umodmsk = 0, modmsk_r = 0;
89  uint64_t val;
90  uint64_t umask;
91  unsigned int modhw = 0;
92  int k, ret, grpid, last_grpid = -1;
93  int grpcounts[INTEL_X86_NUM_GRP];
94  int ncombo[INTEL_X86_NUM_GRP];
95  char umask_str[PFMLIB_EVT_MAX_NAME_LEN];
96 
97  memset(grpcounts, 0, sizeof(grpcounts));
98  memset(ncombo, 0, sizeof(ncombo));
99 
100  umask_str[0] = e->fstr[0] = '\0';
101 
102  reg.val = 0;
103 
104  val = pe[e->event].code;
105 
106  grpmsk = (1 << pe[e->event].ngrp)-1;
107  reg.val |= val; /* preset some filters from code */
108 
109  /* take into account hardcoded umask */
110  umask = (val >> 8) & 0xff;
111 
112  modmsk_r = pe[e->event].modmsk_req;
113 
114  for(k=0; k < e->nattrs; k++) {
115  a = attr(e, k);
116 
117  if (a->ctrl != PFM_ATTR_CTRL_PMU)
118  continue;
119 
120  if (a->type == PFM_ATTR_UMASK) {
121  grpid = pe[e->event].umasks[a->idx].grpid;
122 
123  /*
124  * cfor certain events groups are meant to be
125  * exclusive, i.e., only unit masks of one group
126  * can be used
127  */
128  if (last_grpid != -1 && grpid != last_grpid
129  && intel_x86_eflag(this, e->event, INTEL_X86_GRP_EXCL)) {
130  DPRINT("exclusive unit mask group error\n");
131  return PFM_ERR_FEATCOMB;
132  }
133  /*
134  * upper layer has removed duplicates
135  * so if we come here more than once, it is for two
136  * disinct umasks
137  *
138  * NCOMBO=no combination of unit masks within the same
139  * umask group
140  */
141  ++grpcounts[grpid];
142 
143  if (intel_x86_uflag(this, e->event, a->idx, INTEL_X86_NCOMBO))
144  ncombo[grpid] = 1;
145 
146  if (grpcounts[grpid] > 1 && ncombo[grpid]) {
147  DPRINT("event does not support unit mask combination within a group\n");
148  return PFM_ERR_FEATCOMB;
149  }
150 
151  evt_strcat(umask_str, ":%s", pe[e->event].umasks[a->idx].uname);
152 
153  last_grpid = grpid;
154  modhw |= pe[e->event].umasks[a->idx].modhw;
155  umask |= pe[e->event].umasks[a->idx].ucode >> 8;
156  ugrpmsk |= 1 << pe[e->event].umasks[a->idx].grpid;
157 
158  reg.val |= umask << 8;
159 
160  modmsk_r |= pe[e->event].umasks[a->idx].umodmsk_req;
161 
162  } else if (a->type == PFM_ATTR_RAW_UMASK) {
163 
164  /* there can only be one RAW_UMASK per event */
165 
166  /* sanity check */
167  if (a->idx & ~0xff) {
168  DPRINT("raw umask is 8-bit wide\n");
169  return PFM_ERR_ATTR;
170  }
171  /* override umask */
172  umask = a->idx & 0xff;
173  ugrpmsk = grpmsk;
174  } else {
175  uint64_t ival = e->attrs[k].ival;
176  switch(a->idx) {
177  case NHM_UNC_ATTR_I: /* invert */
178  reg.nhm_unc.usel_inv = !!ival;
179  umodmsk |= _NHM_UNC_ATTR_I;
180  break;
181  case NHM_UNC_ATTR_E: /* edge */
182  reg.nhm_unc.usel_edge = !!ival;
183  umodmsk |= _NHM_UNC_ATTR_E;
184  break;
185  case NHM_UNC_ATTR_C: /* counter-mask */
186  /* already forced, cannot overwrite */
187  if (ival > 255)
188  return PFM_ERR_INVAL;
189  reg.nhm_unc.usel_cnt_mask = ival;
190  umodmsk |= _NHM_UNC_ATTR_C;
191  break;
192  case NHM_UNC_ATTR_O: /* occupancy */
193  reg.nhm_unc.usel_occ = !!ival;
194  umodmsk |= _NHM_UNC_ATTR_O;
195  break;
196  }
197  }
198  }
199 
200  if ((modhw & _NHM_UNC_ATTR_I) && reg.nhm_unc.usel_inv)
201  return PFM_ERR_ATTR_SET;
202  if ((modhw & _NHM_UNC_ATTR_E) && reg.nhm_unc.usel_edge)
203  return PFM_ERR_ATTR_SET;
204  if ((modhw & _NHM_UNC_ATTR_C) && reg.nhm_unc.usel_cnt_mask)
205  return PFM_ERR_ATTR_SET;
206  if ((modhw & _NHM_UNC_ATTR_O) && reg.nhm_unc.usel_occ)
207  return PFM_ERR_ATTR_SET;
208 
209  /*
210  * check that there is at least of unit mask in each unit
211  * mask group
212  */
213  if ((ugrpmsk != grpmsk && !intel_x86_eflag(this, e->event, INTEL_X86_GRP_EXCL)) || ugrpmsk == 0) {
214  ugrpmsk ^= grpmsk;
215  ret = pfm_intel_x86_add_defaults(this, e, ugrpmsk, &umask, (unsigned short) -1, -1);
216  if (ret != PFM_SUCCESS)
217  return ret;
218  }
219 
220  if (modmsk_r && (umodmsk ^ modmsk_r)) {
221  DPRINT("required modifiers missing: 0x%x\n", modmsk_r);
222  return PFM_ERR_ATTR;
223  }
224 
225  evt_strcat(e->fstr, "%s", pe[e->event].name);
226  pfmlib_sort_attr(e);
227  for(k=0; k < e->nattrs; k++) {
228  a = attr(e, k);
229  if (a->ctrl != PFM_ATTR_CTRL_PMU)
230  continue;
231  if (a->type == PFM_ATTR_UMASK)
232  evt_strcat(e->fstr, ":%s", pe[e->event].umasks[a->idx].uname);
233  else if (a->type == PFM_ATTR_RAW_UMASK)
234  evt_strcat(e->fstr, ":0x%x", a->idx);
235  }
236 
237  reg.val |= umask << 8;
238 
239  reg.nhm_unc.usel_en = 1; /* force enable bit to 1 */
240  reg.nhm_unc.usel_int = 1; /* force APIC int to 1 */
241 
242  e->codes[0] = reg.val;
243  e->count = 1;
244 
245  for (k = 0; k < e->npattrs; k++) {
246  int idx;
247 
248  if (e->pattrs[k].ctrl != PFM_ATTR_CTRL_PMU)
249  continue;
250 
251  if (e->pattrs[k].type == PFM_ATTR_UMASK)
252  continue;
253 
254  idx = e->pattrs[k].idx;
255  switch(idx) {
256  case NHM_UNC_ATTR_E:
257  evt_strcat(e->fstr, ":%s=%lu", nhm_unc_mods[idx].name, reg.nhm_unc.usel_edge);
258  break;
259  case NHM_UNC_ATTR_I:
260  evt_strcat(e->fstr, ":%s=%lu", nhm_unc_mods[idx].name, reg.nhm_unc.usel_inv);
261  break;
262  case NHM_UNC_ATTR_C:
263  evt_strcat(e->fstr, ":%s=%lu", nhm_unc_mods[idx].name, reg.nhm_unc.usel_cnt_mask);
264  break;
265  case NHM_UNC_ATTR_O:
266  evt_strcat(e->fstr, ":%s=%lu", nhm_unc_mods[idx].name, reg.nhm_unc.usel_occ);
267  break;
268  }
269  }
270  __pfm_vbprintf("[UNC_PERFEVTSEL=0x%"PRIx64" event=0x%x umask=0x%x en=%d int=%d inv=%d edge=%d occ=%d cnt_msk=%d] %s\n",
271  reg.val,
272  reg.nhm_unc.usel_event,
273  reg.nhm_unc.usel_umask,
274  reg.nhm_unc.usel_en,
275  reg.nhm_unc.usel_int,
276  reg.nhm_unc.usel_inv,
277  reg.nhm_unc.usel_edge,
278  reg.nhm_unc.usel_occ,
279  reg.nhm_unc.usel_cnt_mask,
280  pe[e->event].name);
281 
282  return PFM_SUCCESS;
283 }
