101 static int
102 pfm_rapl_detect(void *this)
103 {
104  int ret;
105 
106  ret = pfm_intel_x86_detect();
107  if (ret != PFM_SUCCESS)
108  return ret;
109 
110  if (pfm_intel_x86_cfg.family != 6)
111  return PFM_ERR_NOTSUPP;
112 
113  switch(pfm_intel_x86_cfg.model) {
114  case 42: /* Sandy Bridge */
115  case 58: /* Ivy Bridge */
116  case 60: /* Haswell */
117  case 69: /* Haswell */
118  case 70: /* Haswell */
119  case 61: /* Broadwell */
120  case 71: /* Broadwell GT3E */
121  case 92: /* Goldmont */
122  case 95: /* Denverton */
123  case 102: /* Cannonlake */
124  case 122: /* Goldmont Plus */
125  /* already setup by default */
126  break;
127  case 45: /* Sandy Bridg-EP */
128  case 62: /* Ivy Bridge-EP */
129  intel_rapl_support.pe = intel_rapl_srv_pe;
130  intel_rapl_support.pme_count = LIBPFM_ARRAY_SIZE(intel_rapl_srv_pe);
131  break;
132  case 78: /* Skylake */
133  case 94: /* Skylake H/S */
134  case 142: /* Kabylake */
135  case 158: /* Kabylake */
136  case 165: /* CometLake mobile */
137  case 166: /* CometLake */
138  case 125: /* Icelake */
139  case 126: /* Icelake mobile */
140  case 157: /* Icelake NNPI */
141  intel_rapl_support.pe = intel_rapl_skl_cln_pe;
142  intel_rapl_support.pme_count = LIBPFM_ARRAY_SIZE(intel_rapl_skl_cln_pe);
143  break;
144  case 63: /* Haswell-EP */
145  case 79: /* Broadwell-EP */
146  case 86: /* Broadwell D */
147  case 85: /* Skylake X */
148  case 106:/* IcelakeX */
149  case 108:/* IcelakeD */
150  case 143:/* SapphireRapidX */
151  intel_rapl_support.pe = intel_rapl_hswep_pe;
152  intel_rapl_support.pme_count = LIBPFM_ARRAY_SIZE(intel_rapl_hswep_pe);
153  break;
154  default :
155  return PFM_ERR_NOTSUPP;
156  }
157  return PFM_SUCCESS;
158 }

160 static int
161 pfm_intel_rapl_get_encoding(void *this, pfmlib_event_desc_t *e)
162 
163 {
164  const intel_x86_entry_t *pe;
165 
166  pe = this_pe(this);
167 
168  e->fstr[0] = '\0';
169 
170  e->codes[0] = pe[e->event].code;
171  e->count = 1;
172  evt_strcat(e->fstr, "%s", pe[e->event].name);
173 
174  __pfm_vbprintf("[0x%"PRIx64" event=0x%x] %s\n",
175  e->codes[0],
176  e->codes[0], e->fstr);
177 
178  return PFM_SUCCESS;
179 }
