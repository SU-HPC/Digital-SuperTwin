135 /*
136  * returns 0 if match (like strcmp())
137  */
138 static int
139 pfm_perf_raw_match_event(void *this, pfmlib_event_desc_t *d, const char *e, const char *s)
140 {
141  uint64_t code;
142  char *endptr = NULL;
143 
144  if (*s != 'r' || !isxdigit(*(s+1)))
145  return 1;
146 
147  code = strtoull(s+1, &endptr, 16);
148  if (code == ULLONG_MAX || errno == ERANGE|| (endptr && *endptr))
149  return 1;
150 
151  /*
152  * stash code in final position
153  */
154  d->codes[0] = code;
155  d->count = 1;
156 
157  return 0;
158 }
