217  void InitMarket(
218  const string& market_filename,
219  ValueT default_value = 1.0,
220  bool verbose = false)
221  {
222  if (verbose) {
223  printf("Reading... "); fflush(stdout);
224  }
225 
226  if (coo_tuples)
227  {
228  fprintf(stderr, "Matrix already constructed\n");
229  exit(1);
230  }
231 
232  std::ifstream ifs;
233  ifs.open(market_filename.c_str(), std::ifstream::in);
234  if (!ifs.good())
235  {
236  fprintf(stderr, "Error opening file\n");
237  exit(1);
238  }
239 
240  bool array = false;
241  bool symmetric = false;
242  bool skew = false;
243  OffsetT current_nz = -1;
244  char line[1024];
245 
246  if (verbose) {
247  printf("Parsing... "); fflush(stdout);
248  }
249 
250  while (true)
251  {
252  ifs.getline(line, 1024);
253  if (!ifs.good())
254  {
255  // Done
256  break;
257  }
258 
259  if (line[0] == '%')
260  {
261  // Comment
262  if (line[1] == '%')
263  {
264  // Banner
265  symmetric = (strstr(line, "symmetric") != NULL);
266  skew = (strstr(line, "skew") != NULL);
267  array = (strstr(line, "array") != NULL);
268 
269  if (verbose) {
270  printf("(symmetric: %d, skew: %d, array: %d) ", symmetric, skew, array); fflush(stdout);
271  }
272  }
273  }
274  else if (current_nz == -1)
275  {
276  // Problem description
277  OffsetT nparsed = sscanf(line, "%d %d %d", &num_rows, &num_cols, &num_nonzeros);
278  if ((!array) && (nparsed == 3))
279  {
280  if (symmetric)
281  num_nonzeros *= 2;
282 
283  // Allocate coo matrix
284  coo_tuples = new CooTuple[num_nonzeros];
285  current_nz = 0;
286 
287  }
288  else if (array && (nparsed == 2))
289  {
290  // Allocate coo matrix
291  num_nonzeros = num_rows * num_cols;
292  coo_tuples = new CooTuple[num_nonzeros];
293  current_nz = 0;
294  }
295  else
296  {
297  fprintf(stderr, "Error parsing MARKET matrix: invalid problem description: %s\n", line);
298  exit(1);
299  }
300 
301  }
302  else
303  {
304  // Edge
305  if (current_nz >= num_nonzeros)
306  {
307  fprintf(stderr, "Error parsing MARKET matrix: encountered more than %d num_nonzeros\n", num_nonzeros);
308  exit(1);
309  }
310 
311  OffsetT row, col;
312  double val;
313 
314  if (array)
315  {
316  if (sscanf(line, "%lf", &val) != 1)
317  {
318  fprintf(stderr, "Error parsing MARKET matrix: badly formed current_nz: '%s' at edge %d\n", line, current_nz);
319  exit(1);
320  }
321  col = (current_nz / num_rows);
322  row = (current_nz - (num_rows * col));
323 
324  coo_tuples[current_nz] = CooTuple(row, col, val); // Convert indices to zero-based
325  }
326  else
327  {
328  // Parse nonzero (note: using strtol and strtod is 2x faster than sscanf or istream parsing)
329  char *l = line;
330  char *t = NULL;
331 
332  // parse row
333  row = strtol(l, &t, 0);
334  if (t == l)
335  {
336  fprintf(stderr, "Error parsing MARKET matrix: badly formed row at edge %d\n", current_nz);
337  exit(1);
338  }
339  l = t;
340 
341  // parse col
342  col = strtol(l, &t, 0);
343  if (t == l)
344  {
345  fprintf(stderr, "Error parsing MARKET matrix: badly formed col at edge %d\n", current_nz);
346  exit(1);
347  }
348  l = t;
349 
350  // parse val
351  val = strtod(l, &t);
352  if (t == l)
353  {
354  val = default_value;
355  }
356 
357  coo_tuples[current_nz] = CooTuple(row - 1, col - 1, val); // Convert indices to zero-based
358  }
359 
360  current_nz++;
361 
362  if (symmetric && (row != col))
363  {
364  coo_tuples[current_nz].row = coo_tuples[current_nz - 1].col;
365  coo_tuples[current_nz].col = coo_tuples[current_nz - 1].row;
366  coo_tuples[current_nz].val = coo_tuples[current_nz - 1].val * (skew ? -1 : 1);
367  current_nz++;
368  }
369  }
370  }
371 
372  // Adjust nonzero count (nonzeros along the diagonal aren't reversed)
373  num_nonzeros = current_nz;
374 
375  if (verbose) {
376  printf("done. "); fflush(stdout);
377  }
378 
379  ifs.close();
380  }

461  int InitGrid2d(OffsetT width, bool self_loop, ValueT default_value = 1.0)
462  {
463  if (coo_tuples)
464  {
465  fprintf(stderr, "Matrix already constructed\n");
466  exit(1);
467  }
468 
469  OffsetT interior_nodes = (width - 2) * (width - 2);
470  OffsetT edge_nodes = (width - 2) * 4;

472  num_rows = width * width;
473  num_cols = num_rows;
474  num_nonzeros = (interior_nodes * 4) + (edge_nodes * 3) + (corner_nodes * 2);
475 
476  if (self_loop)
477  num_nonzeros += num_rows;
478 
479  coo_tuples = new CooTuple[num_nonzeros];
480  OffsetT current_nz = 0;
481 
482  for (OffsetT j = 0; j < width; j++)
483  {
484  for (OffsetT k = 0; k < width; k++)
485  {
486  OffsetT me = (j * width) + k;
487 
488  // West
489  OffsetT neighbor = (j * width) + (k - 1);
490  if (k - 1 >= 0) {
491  coo_tuples[current_nz] = CooTuple(me, neighbor, default_value);
492  current_nz++;
493  }
494 
495  // East
496  neighbor = (j * width) + (k + 1);
497  if (k + 1 < width) {
498  coo_tuples[current_nz] = CooTuple(me, neighbor, default_value);
499  current_nz++;
500  }
501 
502  // North
503  neighbor = ((j - 1) * width) + k;
504  if (j - 1 >= 0) {
505  coo_tuples[current_nz] = CooTuple(me, neighbor, default_value);
506  current_nz++;
507  }
508 
509  // South
510  neighbor = ((j + 1) * width) + k;
511  if (j + 1 < width) {
512  coo_tuples[current_nz] = CooTuple(me, neighbor, default_value);
513  current_nz++;
514  }
515 
516  if (self_loop)
517  {
518  neighbor = me;
519  coo_tuples[current_nz] = CooTuple(me, neighbor, default_value);
520  current_nz++;
521  }
522  }
523  }
524 
525  return 0;

635  // Sort by rows, then columns
636  struct CooComparator
637  {
638  template <typename CooTuple>
639  bool operator()(const CooTuple &a, const CooTuple &b) const
640  {
641  return ((a.row < b.row) || ((a.row == b.row) && (a.col < b.col)));
642  }
643  };

645  OffsetT num_rows;
646  OffsetT num_cols;

648  OffsetT* row_offsets;
649  OffsetT* column_indices;
650  ValueT* values;

653  // Whether to use NUMA malloc to always put storage on the same sockets (for perf repeatability)
654  bool IsNumaMalloc()
655  {
656 #ifdef CUB_MKL
657  return (numa_available() >= 0);
658 #else
659  return false;
660 #endif

666  void Init(
667  CooMatrix<ValueT, OffsetT> &coo_matrix,
668  bool verbose = false)
669  {
670  num_rows = coo_matrix.num_rows;
671  num_cols = coo_matrix.num_cols;
672  num_nonzeros = coo_matrix.num_nonzeros;
673 
674  // Sort by rows, then columns
675  if (verbose) printf("Ordering..."); fflush(stdout);
676  std::stable_sort(coo_matrix.coo_tuples, coo_matrix.coo_tuples + num_nonzeros, CooComparator());
677  if (verbose) printf("done."); fflush(stdout);
678 
679 #ifdef CUB_MKL
680 
681  if (IsNumaMalloc())
682  {
683  numa_set_strict(1);
684 
685  row_offsets = (OffsetT*) numa_alloc_onnode(sizeof(OffsetT) * (num_rows + 1), 0);
686  column_indices = (OffsetT*) numa_alloc_onnode(sizeof(OffsetT) * num_nonzeros, 0);
687 
688  if (numa_num_task_nodes() > 1)
689  values = (ValueT*) numa_alloc_onnode(sizeof(ValueT) * num_nonzeros, 1); // put on different socket than column_indices
690  else
691  values = (ValueT*) numa_alloc_onnode(sizeof(ValueT) * num_nonzeros, 0);
692  }
693  else
694  {
695  values = (ValueT*) mkl_malloc(sizeof(ValueT) * num_nonzeros, 4096);
696  row_offsets = (OffsetT*) mkl_malloc(sizeof(OffsetT) * (num_rows + 1), 4096);
697  column_indices = (OffsetT*) mkl_malloc(sizeof(OffsetT) * num_nonzeros, 4096);
698 
699  }
700 
701 #else
702  row_offsets = new OffsetT[num_rows + 1];
703  column_indices = new OffsetT[num_nonzeros];
704  values = new ValueT[num_nonzeros];
705 #endif
706 
707  OffsetT prev_row = -1;
708  for (OffsetT current_nz = 0; current_nz < num_nonzeros; current_nz++)
709  {
710  OffsetT current_row = coo_matrix.coo_tuples[current_nz].row;
711 
712  // Fill in rows up to and including the current row
713  for (OffsetT row = prev_row + 1; row <= current_row; row++)
714  {
715  row_offsets[row] = current_nz;
716  }
717  prev_row = current_row;
718 
719  column_indices[current_nz] = coo_matrix.coo_tuples[current_nz].col;
720  values[current_nz] = coo_matrix.coo_tuples[current_nz].val;
721  }
722 
723  // Fill out any trailing edgeless vertices (and the end-of-list element)
724  for (OffsetT row = prev_row + 1; row <= num_rows; row++)
725  {
726  row_offsets[row] = num_nonzeros;
727  }
728  }
