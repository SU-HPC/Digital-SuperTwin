62 static int pfm_torrent_detect(void* this)
63 {
64  struct dirent *de;
65  DIR *dir;
66  int ret = PFM_ERR_NOTSUPP;
67 
68  /* If /proc/device-tree/hfi-iohub@<torrent_chip_id> exists,
69  * this machine has an accessible Torrent chip */
70  dir = opendir("/proc/device-tree");
71  if (!dir)
72  return PFM_ERR_NOTSUPP;
73 
74  while ((de = readdir(dir)) != NULL) {
75  if (!strncmp(de->d_name, "hfi-iohub@", 10)) {
76  ret = PFM_SUCCESS;
77  break;
78  }
79  }
80  closedir(dir);
81  return ret;
82 }

149 static int
150 pfm_torrent_get_encoding(void *this, pfmlib_event_desc_t *e)
151 {
152  const pme_torrent_entry_t *pe = this_pe(this);
153  uint32_t torrent_pmu;
154  int i, mod;
155 
156  e->fstr[0] = '\0'; /* initialize the fully-qualified event string */
157 
158  e->count = 1;
159  e->codes[0] = (uint64_t)pe[e->event].pme_code;
160 
161  for (i = 0; i < e->nattrs; i++) {
162 
163  mod = pfm_torrent_attr2mod(this, e->event, e->attrs[i].id);
164  torrent_pmu = pe[e->event].pme_code & (TORRENT_SPACE | TORRENT_PMU_MASK);
165 
166  switch (torrent_pmu) {
167  case TORRENT_PBUS_MCD:
168  switch (mod) {
169  case TORRENT_ATTR_MCD_TYPE:
170  if (e->attrs[i].ival <= 3) {
171  e->codes[0] |= e->attrs[i].ival << TORRENT_ATTR_MCD_TYPE_SHIFT;
172  } else {
173  DPRINT("value of attribute \'type\' - %" PRIu64 " - is not in the range 0..3.\n", e->attrs[i].ival);
174  return PFM_ERR_ATTR_VAL;
175  }
176  break;
177  default:
178  DPRINT("unknown attribute for TORRENT_POWERBUS_MCD - %d\n", mod);
179  return PFM_ERR_ATTR;
180  }
181  break;
182  case TORRENT_PBUS_UTIL:
183  switch (mod) {
184  case TORRENT_ATTR_UTIL_SEL:
185  if (e->attrs[i].ival <= 3) {
186  e->codes[0] |= e->attrs[i].ival << TORRENT_ATTR_UTIL_SEL_SHIFT;
187  } else {
188  DPRINT("value of attribute \'sel\' - %" PRIu64 " - is not in the range 0..3.\n", e->attrs[i].ival);
189  return PFM_ERR_ATTR_VAL;
190  }
191  break;
192  case TORRENT_ATTR_UTIL_LO_CMP:
193  case TORRENT_ATTR_UTIL_HI_CMP:
194  if (e->attrs[i].ival <= 31) {
195  e->codes[0] |= e->attrs[i].ival << TORRENT_ATTR_UTIL_CMP_SHIFT;
196  } else {
197  if (mod == TORRENT_ATTR_UTIL_LO_CMP)
198  DPRINT("value of attribute \'lo_cmp\' - %" PRIu64 " - is not in the range 0..31.\n", e->attrs[i].ival);
199  else
200  DPRINT("value of attribute \'hi_cmp\' - %" PRIu64 " - is not in the range 0..31.\n", e->attrs[i].ival);
201  return PFM_ERR_ATTR_VAL;
202  }
203  }
204  break;
205  default:
206  DPRINT("attributes are unsupported for this Torrent PMU - code = %" PRIx32 "\n", torrent_pmu);
207  return PFM_ERR_ATTR;
208  }
209  }
210  return PFM_SUCCESS;
211 }
