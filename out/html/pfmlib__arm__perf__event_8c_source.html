34 pfm_arm_get_perf_encoding(void *this, pfmlib_event_desc_t *e)
35 {
36  pfmlib_pmu_t *pmu = this;
37  pfm_arm_reg_t reg;
38  struct perf_event_attr *attr = e->os_data;
39  int ret;
40 
41  if (!pmu->get_event_encoding[PFM_OS_NONE])
42  return PFM_ERR_NOTSUPP;
43 
44  /*
45  * use generic raw encoding function first
46  */
47  ret = pmu->get_event_encoding[PFM_OS_NONE](this, e);
48  if (ret != PFM_SUCCESS)
49  return ret;
50 
51  if (e->count > 1) {
52  DPRINT("%s: unsupported count=%d\n", e->count);
53  return PFM_ERR_NOTSUPP;
54  }
55 
56  attr->type = PERF_TYPE_RAW;
57  reg.val = e->codes[0];
58  /*
59  * suppress the bits which are under the control of perf_events.
60  * Recent version of the Linux perf tools may warn if bits which
61  * should not be set by users are set. To avoid the warning,
62  * clear the bits, they are overwritten by the kernel anyway.
63  */
64  reg.evtsel.excl_pl1 = 0;
65  reg.evtsel.excl_usr = 0;
66  reg.evtsel.excl_hyp = 0;
67 
68  attr->config = reg.val;
69  return PFM_SUCCESS;
70 }

72 void
73 pfm_arm_perf_validate_pattrs(void *this, pfmlib_event_desc_t *e)
74 {
75  int i, compact;
76 
77  for (i = 0; i < e->npattrs; i++) {
78  compact = 0;
79 
80  /* umasks never conflict */
81  if (e->pattrs[i].type == PFM_ATTR_UMASK)
82  continue;
83 
84  /*
85  * with perf_events, u and k, hv are handled at the OS
86  * level via attr.exclude_* fields
87  */
88  if (arm_has_plm(this, e) && e->pattrs[i].ctrl == PFM_ATTR_CTRL_PMU) {
89  if ( e->pattrs[i].idx == ARM_ATTR_U
90  || e->pattrs[i].idx == ARM_ATTR_K
91  || e->pattrs[i].idx == ARM_ATTR_HV)
92  compact = 1;
93  }
94  if (e->pattrs[i].ctrl == PFM_ATTR_CTRL_PERF_EVENT) {
95  if (e->pattrs[i].idx == PERF_ATTR_PR)
96  compact = 1;
97  }
98  /* hardware sampling not supported */
99  if (e->pattrs[i].idx == PERF_ATTR_HWS)
100  compact = 1;
101 
102  if (compact) {
103  pfmlib_compact_pattrs(e, i);
104  i--;
105  }
106  }
107 }
