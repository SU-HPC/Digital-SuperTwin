259 /*
260  * allocate space for one new event in event table
261  *
262  * returns NULL if out-of-memory
263  *
264  * may realloc existing table if necessary for growth
265  */
266 static perf_event_t *
267 perf_table_alloc_event(void)
268 {
269  perf_event_t *new_pe;
270  perf_event_t *p;
271 
272  /*
273  * if we need to allocate an event and we have not yet
274  * cloned the static events, then clone them
275  */
276  if (!perf_pe_allocated()) {
277  DPRINT("cloning static event table\n");
278  p = perf_table_clone();
279  if (!p)
280  return NULL;
281  perf_pe = p;
282  }
283 retry:
284  if (perf_pe_free < perf_pe_end)
285  return perf_pe_free++;
286 
287  perf_pe_count += PERF_ALLOC_EVENT_COUNT;
288 
289  new_pe = realloc(perf_pe, perf_pe_count * sizeof(perf_event_t));
290  if (!new_pe)
291  return NULL;
292 
293  perf_pe_free = new_pe + (perf_pe_free - perf_pe);
294  perf_pe_end = perf_pe_free + PERF_ALLOC_EVENT_COUNT;
295  perf_pe = new_pe;
296 
297  goto retry;
298 }

300 /*
301  * allocate space for overflow new unit masks
302  *
303  * Each event can hold up to PERF_MAX_UMASKS.
304  * But gievn we can dynamically add events
305  * which may have more unit masks, then we
306  * put them into a separate overflow unit
307  * masks, table which can grow on demand.
308  * In that case the first PERF_MAX_UMASKS
309  * are in the event, the rest in the overflow
310  * table at index pointed to by event->umask_ovfl_idx
311  * All unit masks for an event are contiguous in the
312  * overflow table.
313  */
314 static perf_umask_t *
315 perf_table_alloc_umask(void)
316 {
317  perf_umask_t *new_um;
318 
319 retry:
320  if (perf_um_free < perf_um_end)
321  return perf_um_free++;
322 
323  perf_um_count += PERF_ALLOC_UMASK_COUNT;
324 
325  new_um = realloc(perf_um, perf_um_count * sizeof(*new_um));
326  if (!new_um)
327  return NULL;
328 
329  perf_um_free = new_um + (perf_um_free - perf_um);
330  perf_um_end = perf_um_free + PERF_ALLOC_UMASK_COUNT;
331  perf_um = new_um;
332 
333  goto retry;
334 }

340 static void
341 gen_tracepoint_table(void)
342 {
343  DIR *dir1, *dir2;
344  struct dirent *d1, *d2;
345  perf_event_t *p;
346  perf_umask_t *um;
347  char d2path[MAXPATHLEN];
348  char idpath[MAXPATHLEN];
349  char id_str[32];
350  uint64_t id;
351  int fd, err;
352  int POTENTIALLY_UNUSED dir2_fd;
353  int reuse_event = 0;
354  int numasks;
355  char *tracepoint_name;
356  int retlen;
357 
358  err = get_debugfs_mnt();
359  if (err == -1)
360  return;
361 
362  strncat(debugfs_mnt, "/tracing/events", MAXPATHLEN-1);
363  debugfs_mnt[MAXPATHLEN-1]= '\0';
364 
365  dir1 = opendir(debugfs_mnt);
366  if (!dir1)
367  return;
368 
369  err = 0;
370  while((d1 = readdir(dir1)) && err >= 0) {
371 
372  if (!strcmp(d1->d_name, "."))
373  continue;
374 
375  if (!strcmp(d1->d_name, ".."))
376  continue;
377 
378  retlen = snprintf(d2path, MAXPATHLEN, "%s/%s", debugfs_mnt, d1->d_name);
379  /* ensure generated d2path string is valid */
380  if (retlen <= 0 || MAXPATHLEN <= retlen)
381  continue;
382 
383  /* fails if d2path is not a directory */
384  dir2 = opendir(d2path);
385  if (!dir2)
386  continue;
387 
388  dir2_fd = dirfd(dir2);
389 
390  /*
391  * if a subdir did not fit our expected
392  * tracepoint format, then we reuse the
393  * allocated space (with have no free)
394  */
395  if (!reuse_event)
396  p = perf_table_alloc_event();
397 
398  if (!p)
399  break;
400 
401  if (p)
402  p->name = tracepoint_name = strdup(d1->d_name);
403 
404  if (!(p && p->name)) {
405  closedir(dir2);
406  err = -1;
407  continue;
408  }
409 
410  p->desc = "tracepoint";
411  p->id = ~0ULL;
412  p->type = PERF_TYPE_TRACEPOINT;
413  p->umask_ovfl_idx = PERF_INVAL_OVFL_IDX;
414  p->modmsk = 0,
415  p->ngrp = 1;
416 
417  numasks = 0;
418  reuse_event = 0;
419 
420  while((d2 = readdir(dir2))) {
421  if (!strcmp(d2->d_name, "."))
422  continue;
423 
424  if (!strcmp(d2->d_name, ".."))
425  continue;
426 
427 #ifdef HAS_OPENAT
428  retlen = snprintf(idpath, MAXPATHLEN, "%s/id", d2->d_name);
429  /* ensure generated d2path string is valid */
430  if (retlen <= 0 || MAXPATHLEN <= retlen)
431  continue;
432 
433  fd = openat(dir2_fd, idpath, O_RDONLY);
434 #else
435  retlen = snprintf(idpath, MAXPATHLEN, "%s/%s/id", d2path, d2->d_name);
436  /* ensure generated d2path string is valid */
437  if (retlen <= 0 || MAXPATHLEN <= retlen)
438  continue;
439 
440  fd = open(idpath, O_RDONLY);
441 #endif
442  if (fd == -1)
443  continue;
444 
445  err = read(fd, id_str, sizeof(id_str));
446 
447  close(fd);
448 
449  if (err < 0)
450  continue;
451 
452  id = strtoull(id_str, NULL, 0);
453 
454  if (numasks < PERF_MAX_UMASKS)
455  um = p->umasks+numasks;
456  else {
457  um = perf_table_alloc_umask();
458  if (numasks == PERF_MAX_UMASKS)
459  p->umask_ovfl_idx = perf_get_ovfl_umask_idx(um);
460  }
461 
462  if (!um) {
463  err = -1;
464  break;
465  }
466 
467  /*
468  * tracepoint have no event codes
469  * the code is in the unit masks
470  */
471  p->id = 0;
472 
473  um->uname = strdup(d2->d_name);
474  if (!um->uname) {
475  err = -1;
476  break;
477  }
478  um->udesc = um->uname;
479  um->uid = id;
480  um->grpid = 0;
481  DPRINT("idpath=%s:%s id=%"PRIu64"\n", p->name, um->uname, id);
482  numasks++;
483  }
484  p->numasks = numasks;
485 
486  closedir(dir2);
487 
488  /*
489  * directory was not pointing
490  * to a tree structure we know about
491  */
492  if (!numasks) {
493  free(tracepoint_name);
494  reuse_event = 1;
495  continue;
496  }
497 
498  /*
499  * update total number of events
500  * only when no error is reported
501  */
502  if (err >= 0)
503  perf_nevents++;
504  reuse_event = 0;
505  }
506  closedir(dir1);
507 }

604 static int
605 pfm_perf_add_defaults(pfmlib_event_desc_t *e, unsigned int msk, uint64_t *umask)
606 {
607  perf_event_t *ent;
608  perf_umask_t *um;
609  int i, j, k, added;
610 
611  k = e->nattrs;
612  ent = perf_pe+e->event;
613 
614  for(i=0; msk; msk >>=1, i++) {
615 
616  if (!(msk & 0x1))
617  continue;
618 
619  added = 0;
620 
621  for(j=0; j < ent->numasks; j++) {
622 
623  if (j < PERF_MAX_UMASKS) {
624  um = &perf_pe[e->event].umasks[j];
625  } else {
626  um = perf_get_ovfl_umask(e->event);
627  um += j - PERF_MAX_UMASKS;
628  }
629  if (um->grpid != i)
630  continue;
631 
632  if (um->uflags & PERF_FL_DEFAULT) {
633  DPRINT("added default %s for group %d\n", um->uname, i);
634 
635  *umask |= um->uid;
636 
637  e->attrs[k].id = j;
638  e->attrs[k].ival = 0;
639  k++;
640 
641  added++;
642  }
643  }
644  if (!added) {
645  DPRINT("no default found for event %s unit mask group %d\n", ent->name, i);
646  return PFM_ERR_UMASK;
647  }
648  }
649  e->nattrs = k;
650  return PFM_SUCCESS;
651 }

654 pfmlib_perf_encode_tp(pfmlib_event_desc_t *e)
655 {

658  int i, nu = 0;

664  * look for tracepoints
665  */

692 static int
693 pfmlib_perf_encode_hw_cache(pfmlib_event_desc_t *e)
694 {
695  pfmlib_event_attr_info_t *a;
696  perf_event_t *ent;
697  unsigned int msk, grpmsk;

702

704 
705  e->codes[0] = ent->id;

707 
708  e->fstr[0] = '\0';
709 
710  for(i=0; i < e->nattrs; i++) {
711  a = attr(e, i);
712  if (a->ctrl != PFM_ATTR_CTRL_PMU)
713  continue;
