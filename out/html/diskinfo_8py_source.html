40 def get_disk_info(name, sizes, hw_lst):
41  hw_lst.append(('disk', name, 'size', str(sizes[name])))
42  info_list = ['device/vendor', 'device/model', 'device/rev',
43  'queue/optimal_io_size', 'queue/physical_block_size',
44  'queue/rotational', 'queue/nr_requests']
45  for info in info_list:
46  disk_sys_info = '/sys/block/%s/%s' % (name, info)
47  # revision can be explicitly named
48  if not os.path.exists(disk_sys_info) and info == 'device/rev':
49  info = 'device/revision'
50  # for nvme devices we can have a nested device dir
51  if not os.path.exists(disk_sys_info):
52  disk_sys_info = '/sys/block/%s/device/%s' % (name, info)
53  try:
54  with open(disk_sys_info, 'r') as dev:
55  hw_lst.append(('disk', name, info.split('/')[1],
56  dev.readline().rstrip('\n').strip()))
57  except Exception as exc:
58  sys.stderr.write(
59  'Failed retrieving disk information %s for %s: %s\n' % (
60  info, name, str(exc)))
61  try:
62  with open('/sys/block/%s/queue/scheduler' % name, 'r') as dev:
63  s_line = dev.readline().rstrip('\n').strip()
64  sched = re.findall(r'\[(.*?)\]', s_line)
65  if sched:
66  hw_lst.append(('disk', name, 'scheduler', sched[0]))
67  except Exception as exc:
68  sys.stderr.write('Cannot extract scheduler for disk %s: %s' % (
69  name, exc))

72 def get_disk_cache(name, hw_lst):
73  # WCE & RCD from sysfs
74  # https://www.kernel.org/doc/Documentation/scsi/sd-parameters.txt
75  device_path = '/sys/block/{0}/device'.format(name)
76 
77  try:
78  _link_info = os.readlink(device_path)
79  _scsi_addr = _link_info.rsplit('/', 1)[1]
80  device_path = (device_path + '/scsi_disk/{0}/cache_type').format(
81  _scsi_addr)
82  with open(device_path, 'r') as cache_info:
83  my_text = cache_info.readline().rstrip('\n').strip()
84  _wce = '1'
85  _rcd = '0'
86  if my_text == 'write through':
87  _wce = '0'
88  elif my_text == 'none':
89  _wce = '0'
90  _rcd = '1'
91  elif 'daft' in my_text:
92  _rcd = '1'
93  hw_lst.append(('disk', name, 'Write Cache Enable', _wce))
94  hw_lst.append(('disk', name, 'Read Cache Disable', _rcd))
95  except Exception as exc:
96  sys.stderr.write(
97  'Failed at getting disk information at %s: %s\n' %
98  (device_path, str(exc)))

101 def get_disk_id(name, hw_lst):
102  # In some VMs, the disk-by id doesn't exists
103  if os.path.exists('/dev/disk/by-id/'):
104  for entry in os.listdir('/dev/disk/by-id/'):
105  idp = os.path.realpath('/dev/disk/by-id/' + entry).split('/')
106  if idp[-1] == name:
107  id_name = "id"
108  if entry.startswith('wwn'):
109  id_name = "wwn-id"
110  elif entry.startswith('scsi'):
111  id_name = "scsi-id"
112  hw_lst.append(('disk', name, id_name, entry))

135 def detect():
136  """Detect disks."""
137 
138  hw_lst = []
139  names = disknames()
140  sizes = disksizes(names)
141  disks = [name for name, size in sizes.items() if size > 0]
142  hw_lst.append(('disk', 'logical', 'count', str(len(disks))))
143  for name in disks:
144  get_disk_info(name, sizes, hw_lst)
145 
146  # nvme devices do not need standard cache mechanisms
147  if not name.startswith('nvme'):
148  get_disk_cache(name, hw_lst)
149 
150  get_disk_id(name, hw_lst)
151 
152  # smartctl support
153  # run only if smartctl command is there
154  if detect_utils.which("smartctl"):
155  if name.startswith('nvme'):
156  sys.stderr.write('Reading SMART for nvme\n')
157  smart_utils.read_smart_nvme(hw_lst, name)
158  else:
159  smart_utils.read_smart(hw_lst, "/dev/%s" % name)
160  else:
161  sys.stderr.write("Cannot find smartctl, exiting\n")
162 
163  return hw_lst
