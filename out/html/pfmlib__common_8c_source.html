736 /*
737  * pfmlib_getl: our own equivalent to GNU getline() extension.
738  * This avoids a dependency on having a C library with
739  * support for getline().
740  */
741 int
742 pfmlib_getl(char **buffer, size_t *len, FILE *fp)
743 {
744 #define GETL_DFL_LEN 32
745  char *b;
746  int c;
747  size_t maxsz, maxi, d, i = 0;
748 
749  if (!len || !fp || !buffer || (*buffer && *len < 2))
750  return -1;
751 
752  b = *buffer;
753 
754  if (!b)
755  *len = 0;
756 
757  maxsz = *len;
758  maxi = maxsz - 2;
759 
760  while ((c = fgetc(fp)) != EOF) {
761  if (maxsz == 0 || i == maxi) {
762  if (maxsz == 0)
763  maxsz = GETL_DFL_LEN;
764  else
765  maxsz <<= 1;
766 
767  if (*buffer)
768  d = &b[i] - *buffer;
769  else
770  d = 0;
771 
772  *buffer = realloc(*buffer, maxsz);
773  if (!*buffer)
774  return -1;
775 
776  b = *buffer + d;
777  maxi = maxsz - d - 2;
778  i = 0;
779  *len = maxsz;
780  }
781  b[i++] = c;
782  if (c == '\n')
783  break;
784  }
785  b[i] = '\0';
786  return c != EOF ? 0 : -1;
787 }

942 {
943  size_t rsz = sz;
944 
945  /*
946  * if user size is zero, then use ABI0 size
947  */
948  if (usz == 0)
949  usz = refsz;
950 
951  /*
952  * cannot be smaller than ABI0 size
953  */

955  DPRINT("pfmlib_check_struct: user size too small %zu\n", usz);
956  return 0;
957  }
958

960  * if bigger than current ABI, then check that none
961  * of the extra bits are set. This is to avoid mistake
962  * by caller assuming the library set those bits.

966  char *end = (char *)st + usz;

968  if (*addr++) {
969  DPRINT("pfmlib_check_struct: invalid extra bits\n");
970  return 0;
971  }
972  }
973  }
974  return rsz;
975 }

1307 static int
1308 pfmlib_parse_event_attr(char *str, pfmlib_event_desc_t *d)
1309 {
1310  pfmlib_event_attr_info_t *ainfo;
1311  char *s, *p, *q, *endptr;
1312  char yes[2] = "y";
1313  pfm_attr_t type;
1314  int aidx = 0, has_val, has_raw_um = 0, has_um = 0;
1315  int ret = PFM_ERR_INVAL;
1316 
1317  s = str;
1318 
1319  while(s) {
1320  p = s;
1321  pfmlib_strsep(&p, PFMLIB_ATTR_DELIM);
1322  /* if (p)
1323  *p++ = '\0'; */
1324 
1325  q = strchr(s, '=');
1326  if (q)
1327  *q++ = '\0';
1328 
1329  has_val = !!q;
1330 
1331  /*
1332  * check for raw umasks in hexdecimal only
1333  */
1334  if (*s == '0' && tolower(*(s+1)) == 'x') {
1335  char *endptr = NULL;
1336 
1337  /* can only have one raw umask */
1338  if (has_raw_um || has_um) {
1339  DPRINT("cannot mix raw umask with umask\n");
1340  return PFM_ERR_ATTR;
1341  }
1342  if (!(d->pmu->flags & PFMLIB_PMU_FL_RAW_UMASK)) {
1343  DPRINT("PMU %s does not support RAW umasks\n", d->pmu->name);
1344  return PFM_ERR_ATTR;
1345  }
1346 
1347  /* we have reserved an entry at the end of pattrs */
1348  aidx = d->npattrs;
1349  ainfo = d->pattrs + aidx;
1350 
1351  ainfo->name = "RAW_UMASK";
1352  ainfo->type = PFM_ATTR_RAW_UMASK;
1353  ainfo->ctrl = PFM_ATTR_CTRL_PMU;
1354  /* can handle up to 64-bit raw umask */
1355  ainfo->idx = strtoull(s, &endptr, 0);
1356  ainfo->equiv= NULL;
1357  if (*endptr) {
1358  DPRINT("raw umask (%s) is not a number\n");
1359  return PFM_ERR_ATTR;
1360  }
1361 
1362  has_raw_um = 1;
1363 
1364  goto found_attr;
1365  }
1366 
1367  for(aidx = 0; aidx < d->npattrs; aidx++) {
1368  if (!strcasecmp(d->pattrs[aidx].name, s)) {
1369  ainfo = d->pattrs + aidx;
1370  /* disambiguate modifier and umask
1371  * with the same name : snb::L2_LINES_IN:I:I=1
1372  */
1373  if (has_val && ainfo->type == PFM_ATTR_UMASK)
1374  continue;
1375  goto found_attr;
1376  }
1377  }
1378  DPRINT("cannot find attribute %s\n", s);
1379  return PFM_ERR_ATTR;
1380 found_attr:
1381  type = ainfo->type;
1382 
1383  if (type == PFM_ATTR_UMASK) {
1384  has_um = 1;
1385  if (has_raw_um) {
1386  DPRINT("cannot mix raw umask with umask\n");
1387  return PFM_ERR_ATTR;
1388  }
1389  }
1390 
1391  if (ainfo->equiv) {
1392  char *z;
1393 
1394  /* cannot have equiv for attributes with value */
1395  if (has_val)
1396  return PFM_ERR_ATTR_VAL;
1397 
1398  /* copy because it is const */
1399  z = strdup(ainfo->equiv);
1400  if (!z)
1401  return PFM_ERR_NOMEM;
1402 
1403  ret = pfmlib_parse_event_attr(z, d);
1404 
1405  free(z);
1406 
1407  if (ret != PFM_SUCCESS)
1408  return ret;
1409  s = p;
1410  continue;
1411  }
1412  /*
1413  * we tolerate missing value for boolean attributes.
1414  * Presence of the attribute is equivalent to
1415  * attr=1, i.e., attribute is set
1416  */
1417  if (type != PFM_ATTR_UMASK && type != PFM_ATTR_RAW_UMASK && !has_val) {
1418  if (type != PFM_ATTR_MOD_BOOL)
1419  return PFM_ERR_ATTR_VAL;
1420  s = yes; /* no const */
1421  goto handle_bool;
1422  }
1423 
1424  d->attrs[d->nattrs].ival = 0;
1425  if ((type == PFM_ATTR_UMASK || type == PFM_ATTR_RAW_UMASK) && has_val)
1426  return PFM_ERR_ATTR_VAL;
1427 
1428  if (has_val) {
1429  s = q;
1430 handle_bool:
1431  ret = PFM_ERR_ATTR_VAL;
1432  if (!strlen(s))
1433  goto error;
1434  if (d->nattrs == PFMLIB_MAX_ATTRS) {
1435  DPRINT("too many attributes\n");
1436  ret = PFM_ERR_TOOMANY;
1437  goto error;
1438  }
1439 
1440  endptr = NULL;
1441  switch(type) {
1442  case PFM_ATTR_MOD_BOOL:
1443  if (strlen(s) > 1)
1444  goto error;
1445 
1446  if (tolower((int)*s) == 'y'
1447  || tolower((int)*s) == 't' || *s == '1')
1448  d->attrs[d->nattrs].ival = 1;
1449  else if (tolower((int)*s) == 'n'
1450  || tolower((int)*s) == 'f' || *s == '0')
1451  d->attrs[d->nattrs].ival = 0;
1452  else
1453  goto error;
1454  break;
1455  case PFM_ATTR_MOD_INTEGER:
1456  d->attrs[d->nattrs].ival = strtoull(s, &endptr, 0);
1457  if (*endptr != '\0')
1458  goto error;
1459  break;
1460  default:
1461  goto error;
1462  }
1463  }
1464  d->attrs[d->nattrs].id = aidx;
1465  d->nattrs++;
1466  s = p;
1467  }
1468  ret = PFM_SUCCESS;
1469 error:
1470  return ret;
1471 }

1609 int
1610 pfmlib_parse_event(const char *event, pfmlib_event_desc_t *d)
1611 {
1612  pfm_event_info_t einfo;
1613  char *str, *s, *p;
1614  pfmlib_pmu_t *pmu;
1615  int (*match)(void *this, pfmlib_event_desc_t *d, const char *e, const char *s);
1616  const char *pname = NULL;
1617  int i, j, ret;
1618 
1619  /*
1620  * support only one event at a time.
1621  */
1622  p = strpbrk(event, PFMLIB_EVENT_DELIM);
1623  if (p)
1624  return PFM_ERR_INVAL;
1625  /*
1626  * create copy because string is const
1627  */
1628  s = str = strdup(event);
1629  if (!str)
1630  return PFM_ERR_NOMEM;
1631 
1632 
1633  /* check for optional PMU name */
1634  p = strstr(s, PFMLIB_PMU_DELIM);
1635  if (p) {
1636  *p = '\0';
1637  pname = s;
1638  s = p + strlen(PFMLIB_PMU_DELIM);
1639  }
1640  p = s;
1641  pfmlib_strsep(&p, PFMLIB_ATTR_DELIM);
1642  /* if (p)
1643  *p++ = '\0'; */
1644 
1645  /*
1646  * for each pmu
1647  */
1648  pfmlib_for_each_pmu(j) {
1649  pmu = pfmlib_pmus[j];
1650  /*
1651  * if no explicit PMU name is given, then
1652  * only look for active PMU models
1653  */
1654  if (!pname && !pfmlib_pmu_active(pmu))
1655  continue;
1656 
1657  /*
1658  * if the PMU name is not passed, then if
1659  * the pmu is deprecated, then skip it. It means
1660  * there is a better candidate in the active list
1661  */
1662  if (!pname && pfmlib_pmu_deprecated(pmu))
1663  continue;
1664  /*
1665  * check for requested PMU name,
1666  */
1667  if (pname && strcasecmp(pname, pmu->name))
1668  continue;
1669  /*
1670  * only allow event on inactive PMU if enabled via
1671  * environement variable
1672  */
1673  if (pname && !pfmlib_pmu_active(pmu) && !pfm_cfg.inactive)
1674  continue;
1675 
1676  match = pmu->match_event ? pmu->match_event : match_event;
1677  /*
1678  * for each event
1679  */
1680  pfmlib_for_each_pmu_event(pmu, i) {
1681  ret = pmu->get_event_info(pmu, i, &einfo);
1682  if (ret != PFM_SUCCESS)
1683  goto error;
1684  if (!match(pmu, d, einfo.name, s))
1685  goto found;
1686  }
1687  }
1688  free(str);
1689  return PFM_ERR_NOTFOUND;
1690 found:
1691  d->pmu = pmu;
1692  /*
1693  * handle equivalence
1694  */
1695  if (einfo.equiv) {
1696  ret = pfmlib_parse_equiv_event(einfo.equiv, d);
1697  if (ret != PFM_SUCCESS)
1698  goto error;
1699  } else {
1700  d->event = i; /* private index */
1701 
1702  ret = pfmlib_build_event_pattrs(d);
1703  if (ret != PFM_SUCCESS)
1704  goto error;
1705  }
1706  /*
1707  * parse attributes from original event
1708  */
1709  ret = pfmlib_parse_event_attr(p, d);
1710  if (ret == PFM_SUCCESS)
1711  ret = pfmlib_sanitize_event(d);
1712 
1713  for (i = 0; i < d->nattrs; i++) {
1714  pfmlib_event_attr_info_t *a = attr(d, i);
1715  if (a->type != PFM_ATTR_RAW_UMASK)
1716  DPRINT("%d %d %d %s\n", d->event, i, a->idx, d->pattrs[d->attrs[i].id].name);
1717  else
1718  DPRINT("%d %d RAW_UMASK (0x%x)\n", d->event, i, a->idx);
1719  }
1720 error:
1721  free(str);
1722  if (ret != PFM_SUCCESS)
1723  pfmlib_release_event(d);
1724  return ret;
1725 }
