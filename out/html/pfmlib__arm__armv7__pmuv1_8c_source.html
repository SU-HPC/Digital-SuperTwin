43 pfm_arm_detect_cortex_a7(void *this)
44 {
45 
46  int ret;
47 
48  ret = pfm_arm_detect(this);
49  if (ret != PFM_SUCCESS)
50  return PFM_ERR_NOTSUPP;

53  (pfm_arm_cfg.part == 0xc07)) { /* Cortex-A7 */
54  return PFM_SUCCESS;

56  return PFM_ERR_NOTSUPP;

110 static int
111 pfm_arm_detect_krait(void *this)
112 {
113 
114  int ret;
115 
116  ret = pfm_arm_detect(this);
117  if (ret != PFM_SUCCESS)
118  return PFM_ERR_NOTSUPP;
119 
120  /* Check for Qualcomm */
121  if (pfm_arm_cfg.implementer == 0x51) {
122  /* Check that [15:10] of midr is 0x01 which */
123  /* indicates Krait rather than Scorpion CPU */
124  /* pfm_arm_cfg.part is (midr>>4)&0xfff */
125  if (pfm_arm_cfg.part >> 6 == 0x1) {
126  return PFM_SUCCESS;
127  }
128  }
129  return PFM_ERR_NOTSUPP;
130 }
