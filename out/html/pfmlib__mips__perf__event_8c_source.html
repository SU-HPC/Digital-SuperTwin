32 int
33 pfm_mips_get_perf_encoding(void *this, pfmlib_event_desc_t *e)
34 {
35  struct perf_event_attr *attr = e->os_data;
36  int ret;
37 
38  ret = pfm_mips_get_encoding(this, e);
39  if (ret != PFM_SUCCESS)
40  return ret;
41 
42  if (e->count != 2) {
43  DPRINT("unexpected encoding count=%d\n", e->count);
44  return PFM_ERR_INVAL;
45  }
46  attr->type = PERF_TYPE_RAW;
47 
48  /*
49  * priv levels are ignored because they are managed
50  * directly through perf excl_*.
51  */
52  attr->config = e->codes[0] >> 5;
53 
54  /*
55  * codes[1] contains counter mask supported by the event.
56  * Events support either odd or even indexed counters
57  * except for cycles (code = 0) and instructions (code =1)
58  * which work on all counters.
59  *
60  * The kernel expects bit 7 of config to indicate whether
61  * the event works only on odd-indexed counters
62  */
63  if ((e->codes[1] & 0x2) && attr->config > 1)
64  attr->config |= 1ULL << 7;
65 
66  return PFM_SUCCESS;
67 }

69 void
70 pfm_mips_perf_validate_pattrs(void *this, pfmlib_event_desc_t *e)
71 {
72  int i, compact;
73 
74  for (i = 0; i < e->npattrs; i++) {
75  compact = 0;
76 
77  /* umasks never conflict */
78  if (e->pattrs[i].type == PFM_ATTR_UMASK)
79  continue;
80 
81  /*
82  * remove PMU-provided attributes which are either
83  * not accessible under perf_events or fully controlled
84  * by perf_events, e.g., priv levels filters
85  */
86  if (e->pattrs[i].ctrl == PFM_ATTR_CTRL_PMU) {
87  /*
88  * with perf_event, priv levels under full
89  * control of perf_event.
90  */
91  if ( e->pattrs[i].idx == MIPS_ATTR_K
92  ||e->pattrs[i].idx == MIPS_ATTR_U
93  ||e->pattrs[i].idx == MIPS_ATTR_S
94  ||e->pattrs[i].idx == MIPS_ATTR_E)
95  compact = 1;
96  }
97  /*
98  * remove perf_event generic attributes not supported
99  * by MIPS
100  */
101  if (e->pattrs[i].ctrl == PFM_ATTR_CTRL_PERF_EVENT) {
102  /* no precise sampling on MIPS */
103  if (e->pattrs[i].idx == PERF_ATTR_PR)
104  compact = 1;
105  }
106 
107  /* hardware sampling not supported */
108  if (e->pattrs[i].idx == PERF_ATTR_HWS)
109  compact = 1;
110 
111 
112  if (compact) {
113  pfmlib_compact_pattrs(e, i);
114  i--;
115  }
116  }
117 }
