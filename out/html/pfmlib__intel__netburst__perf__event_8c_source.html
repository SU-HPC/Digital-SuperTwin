65 pfm_netburst_perf_validate_pattrs(void *this, pfmlib_event_desc_t *e)
66 {
67  int i, compact;
68 
69  for (i = 0; i < e->npattrs; i++) {
70  compact = 0;
71 
72  /* umasks never conflict */
73  if (e->pattrs[i].type == PFM_ATTR_UMASK)
74  continue;
75 
76  /*
77  * with perf_events, u and k are handled at the OS level
78  * via exclude_user, exclude_kernel.
79  */
80  if (e->pattrs[i].ctrl == PFM_ATTR_CTRL_PMU) {
81  if (e->pattrs[i].idx == NETBURST_ATTR_U
82  || e->pattrs[i].idx == NETBURST_ATTR_K)
83  compact = 1;
84  }
85  if (e->pattrs[i].ctrl == PFM_ATTR_CTRL_PERF_EVENT) {
86 
87  /* no PEBS support (for now) */
88  if (e->pattrs[i].idx == PERF_ATTR_PR)
89  compact = 1;
90  /*
91  * No hypervisor on Intel */
92  if (e->pattrs[i].idx == PERF_ATTR_H)
93  compact = 1;
94  }
95 
96  /* hardware sampling not supported */
97  if (e->pattrs[i].idx == PERF_ATTR_HWS)
98  compact = 1;
99 
100 
101  if (compact) {
102  pfmlib_compact_pattrs(e, i);
103  i--;
104  }
105  }
