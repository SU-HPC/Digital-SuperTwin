86 inline int ComputeRankOfMatrixRow(const Geometry & geom, global_int_t index) {
87  global_int_t gnx = geom.gnx;
88  global_int_t gny = geom.gny;
89 
90  global_int_t iz = index/(gny*gnx);
91  global_int_t iy = (index-iz*gny*gnx)/gnx;
92  global_int_t ix = index%gnx;
93  // We now permit varying values for nz for any nx-by-ny plane of MPI processes.
94  // npartz is the number of different groups of nx-by-ny groups of processes.
95  // partz_ids is an array of length npartz where each value indicates the z process of the last process in the ith nx-by-ny group.
96  // partz_nz is an array of length npartz containing the value of nz for the ith group.
97 
98  // With no variation, npartz = 1, partz_ids[0] = npz, partz_nz[0] = nz
99 
100  int ipz = 0;
101  int ipartz_ids = 0;
102  for (int i=0; i< geom.npartz; ++i) {
103  int ipart_nz = geom.partz_nz[i];
104  ipartz_ids = geom.partz_ids[i] - ipartz_ids;
105  if (iz<= ipart_nz*ipartz_ids) {
106  ipz += iz/ipart_nz;
107  break;
108  } else {
109  ipz += ipartz_ids;
110  iz -= ipart_nz*ipartz_ids;
111  }
112 
113  }
114 // global_int_t ipz = iz/geom.nz;
115  int ipy = iy/geom.ny;
116  int ipx = ix/geom.nx;
117  int rank = ipx+ipy*geom.npx+ipz*geom.npy*geom.npx;
118  return rank;
119 }
