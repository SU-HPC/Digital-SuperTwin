35 find_pmu_type_by_name(const char *name)
36 {
37  char filename[PATH_MAX];
38  FILE *fp;
39  int ret, type;
40 
41  if (!name)
42  return PFM_ERR_NOTSUPP;
43 
44  sprintf(filename, "/sys/bus/event_source/devices/%s/type", name);
45 
46  fp = fopen(filename, "r");
47  if (!fp)
48  return PFM_ERR_NOTSUPP;
49 
50  ret = fscanf(fp, "%d", &type);
51  if (ret != 1)
52  type = PFM_ERR_NOTSUPP;
53 
54  fclose(fp);
55 
56  return type;
57 }

59 int
60 pfm_amd64_get_perf_encoding(void *this, pfmlib_event_desc_t *e)
61 {
62  pfmlib_pmu_t *pmu = this;
63  pfm_amd64_reg_t reg;
64  struct perf_event_attr *attr = e->os_data;
65  int ret;
66 
67  if (!pmu->get_event_encoding[PFM_OS_NONE])
68  return PFM_ERR_NOTSUPP;
69 
70  /*
71  * use generic raw encoding function first
72  */
73  ret = pmu->get_event_encoding[PFM_OS_NONE](this, e);
74  if (ret != PFM_SUCCESS)
75  return ret;
76 
77  if (e->count > 1) {
78  DPRINT("%s: unsupported count=%d\n", e->count);
79  return PFM_ERR_NOTSUPP;
80  }
81 
82  ret = PERF_TYPE_RAW;
83 
84  /*
85  * if specific perf PMU is provided then try to locate it
86  * otherwise assume core PMU and thus type RAW
87  */
88  if (pmu->perf_name) {
89  /* greab PMU type from sysfs */
90  ret = find_pmu_type_by_name(pmu->perf_name);
91  if (ret < 0)
92  return ret;
93  }
94  DPRINT("amd64_get_perf_encoding: PMU type=%d\n", ret);
95  attr->type = ret;
96 
97  reg.val = e->codes[0];
98  /*
99  * suppress the bits which are under the control of perf_events
100  * they will be ignore by the perf tool and the kernel interface
101  * the OS/USR bits are controlled by the attr.exclude_* fields
102  * the EN/INT bits are controlled by the kernel
103  */
104  reg.sel_en = 0;
105  reg.sel_int = 0;
106  reg.sel_os = 0;
107  reg.sel_usr = 0;
108  reg.sel_guest = 0;
109  reg.sel_host = 0;
110 
111  attr->config = reg.val;
112 
113  return PFM_SUCCESS;
114 }

116 void
117 pfm_amd64_perf_validate_pattrs(void *this, pfmlib_event_desc_t *e)
118 {
119  pfmlib_pmu_t *pmu = this;
120 
121  int i, compact;
122 
123  for (i=0; i < e->npattrs; i++) {
124  compact = 0;
125 
126  /* umasks never conflict */
127  if (e->pattrs[i].type == PFM_ATTR_UMASK)
128  continue;
129 
130  /*
131  * with perf_events, u and k are handled at the OS level
132  * via attr.exclude_* fields
133  */
134  if (e->pattrs[i].ctrl == PFM_ATTR_CTRL_PMU) {
135 
136  if (e->pattrs[i].idx == AMD64_ATTR_U
137  || e->pattrs[i].idx == AMD64_ATTR_K
138  || e->pattrs[i].idx == AMD64_ATTR_H)
139  compact = 1;
140  }
141 
142  if (e->pattrs[i].ctrl == PFM_ATTR_CTRL_PERF_EVENT) {
143 
144  /* No precise mode on AMD */
145  if (e->pattrs[i].idx == PERF_ATTR_PR)
146  compact = 1;
147 
148  /* older processors do not support hypervisor priv level */
149  if (e->pattrs[i].idx == PERF_ATTR_H &&!pfm_amd64_supports_virt(pmu))
150  compact = 1;
151  }
152 
153  /* hardware sampling not supported */
154  if (e->pattrs[i].idx == PERF_ATTR_HWS)
155  compact = 1;
156 
157  if (compact) {
158  pfmlib_compact_pattrs(e, i);
159  i--;
160  }
161  }
162 }
