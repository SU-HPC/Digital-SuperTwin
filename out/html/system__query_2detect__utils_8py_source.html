47 def parse_lldtool(hw_lst, interface_name, lines):
48  content = ""
49  header = ""
50  sub_header = ""
51  for line in lines:
52  if line.startswith('\t'):
53  content = line.strip().strip('\n').strip('\t').replace("/", "_")
54  else:
55  header = line
56  header = line.strip().strip('\n').strip('\t').replace("/", "_")
57  header = header.replace(" TLV", "")
58  content = ""
59  sub_header = ""
60  if header and content:
61  if ":" in content:
62  line = content.split(":")
63  if (len(line) == 2) and (line[1] == ''):
64  sub_header = line[0].strip().strip('\n').strip('\t')
65  sub_header = sub_header.replace("/", "_")
66  sub_header = sub_header.replace(" TLV:", "")
67  header = header + "/" + sub_header
68  continue
69  else:
70  left = line[0].strip().strip('\n').strip('\t')
71  left = left.replace("/", "_")
72  right = content.replace(left + ":", "").strip().strip('\n')
73  right = right.strip('\t').replace("/", "_")
74  # If we never had this sub_header for this header
75  # let's add one
76  if left != sub_header:
77  sub_header = left
78  header = header + "/" + sub_header
79  content = right
80  hw_lst.append(('lldp', interface_name, header, content))
81 
82  return hw_lst

179 def _get_uuid_ppc64le(hw_lst):
180  """Get uuid for ppc64 arch systems"""
181  vendor = None
182  serial = None
183  for (_, _, sys_subtype, value) in hw_lst:
184  if sys_subtype == 'vendor':
185  vendor = value
186  if sys_subtype == 'serial':
187  serial = value
188 
189  system_uuid = None
190  system_uuid_fname = '/sys/firmware/devicetree/base/system-uuid'
191  if os.access(system_uuid_fname, os.R_OK):
192  with open(system_uuid_fname) as uuidfile:
193  system_uuid = uuidfile.read().rstrip(' \t\r\n\0')
194  elif vendor and serial:
195  root = uuid.UUID(bytes=b'\x00' * 16)
196  vendor_uuid = uuid.uuid5(root, vendor)
197  system_uuid = str(uuid.uuid5(vendor_uuid, serial))
198 
199  return system_uuid
