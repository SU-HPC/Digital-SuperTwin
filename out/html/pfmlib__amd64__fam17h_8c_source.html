32 /*
33  * This function detects ZEN1 for the deprecated
34  * amd_fam17h pmu model name.
35  */
36 static int
37 pfm_amd64_family_detect_zen1(void *this)
38 {
39  int ret, rev;
40 
41  ret = pfm_amd64_detect(this);
42  if (ret != PFM_SUCCESS)
43  return ret;
44 
45  rev = pfm_amd64_cfg.revision;
46  return rev == PFM_PMU_AMD64_FAM17H_ZEN1 ? PFM_SUCCESS: PFM_ERR_NOTSUPP;
47 }
