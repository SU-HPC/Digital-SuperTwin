127 /*
128  * detect processor model using cpuid()
129  * based on documentation
130  * http://www.intel.com/Assets/PDF/appnote/241618.pdf
131  */
132 int
133 pfm_intel_x86_detect(void)
134 {
135  unsigned int a, b, c, d;
136  char buffer[64];
137 
138  if (pfm_intel_x86_cfg.family)
139  return PFM_SUCCESS;
140 
141  cpuid(0, &a, &b, &c, &d);
142  strncpy(&buffer[0], (char *)(&b), 4);
143  strncpy(&buffer[4], (char *)(&d), 4);
144  strncpy(&buffer[8], (char *)(&c), 4);
145  buffer[12] = '\0';
146 
147  /* must be Intel */
148  if (strcmp(buffer, "GenuineIntel"))
149  return PFM_ERR_NOTSUPP;
150 
151  cpuid(1, &a, &b, &c, &d);
152 
153  pfm_intel_x86_cfg.family = (a >> 8) & 0xf; // bits 11 - 8
154  pfm_intel_x86_cfg.model = (a >> 4) & 0xf; // Bits 7 - 4
155  pfm_intel_x86_cfg.stepping = a & 0xf; // Bits 0 - 3
156 
157  /* extended family */
158  if (pfm_intel_x86_cfg.family == 0xf)
159  pfm_intel_x86_cfg.family += (a >> 20) & 0xff;
160 
161  /* extended model */
162  if (pfm_intel_x86_cfg.family >= 0x6)
163  pfm_intel_x86_cfg.model += ((a >> 16) & 0xf) << 4;
164 
165  return PFM_SUCCESS;
166 }

188 int
189 pfm_intel_x86_add_defaults(void *this, pfmlib_event_desc_t *e,
190  unsigned int msk,
191  uint64_t *umask,
192  unsigned short max_grpid,
193  int excl_grp_but_0)
194 {
195  const intel_x86_entry_t *pe = this_pe(this);
196  const intel_x86_entry_t *ent;
197  unsigned int i;
198  unsigned short grpid;
199  int j, k, added, skip;
200  int idx;
201 
202  k = e->nattrs;
203  ent = pe+e->event;
204 
205  for(i=0; msk; msk >>=1, i++) {
206 
207  if (!(msk & 0x1))
208  continue;
209 
210  added = skip = 0;
211  /*
212  * must scan list of possible attributes
213  * (not all possible attributes)
214  */
215  for (j = 0; j < e->npattrs; j++) {
216  if (e->pattrs[j].ctrl != PFM_ATTR_CTRL_PMU)
217  continue;
218 
219  if (e->pattrs[j].type != PFM_ATTR_UMASK)
220  continue;
221 
222  idx = e->pattrs[j].idx;
223 
224  if (get_grpid(ent->umasks[idx].grpid) != i)
225  continue;
226 
227  if (max_grpid != INTEL_X86_MAX_GRPID && i > max_grpid) {
228  skip = 1;
229  continue;
230  }
231 
232  if (intel_x86_uflag(this, e->event, idx, INTEL_X86_GRP_DFL_NONE)) {
233  skip = 1;
234  continue;
235  }
236  grpid = ent->umasks[idx].grpid;
237 
238  if (excl_grp_but_0 != -1 && grpid != 0 && excl_grp_but_0 != grpid) {
239  skip = 1;
240  continue;
241  }
242 
243  /* umask is default for group */
244  if (intel_x86_uflag(this, e->event, idx, INTEL_X86_DFL)) {
245  DPRINT("added default %s for group %d j=%d idx=%d ucode=0x%"PRIx64"\n",
246  ent->umasks[idx].uname,
247  i,
248  j,
249  idx,
250  ent->umasks[idx].ucode);
251  /*
252  * default could be an alias, but
253  * ucode must reflect actual code
254  */
255  *umask |= ent->umasks[idx].ucode >> 8;
256 
257  e->attrs[k].id = j; /* pattrs index */
258  e->attrs[k].ival = 0;
259  k++;
260 
261  added++;
262  if (intel_x86_eflag(this, e->event, INTEL_X86_GRP_EXCL))
263  goto done;
264 
265  if (intel_x86_uflag(this, e->event, idx, INTEL_X86_EXCL_GRP_GT)) {
266  if (max_grpid != INTEL_X86_MAX_GRPID) {
267  DPRINT("two max_grpid, old=%d new=%d\n", max_grpid, get_grpid(ent->umasks[idx].grpid));
268  return PFM_ERR_UMASK;
269  }
270  max_grpid = ent->umasks[idx].grpid;
271  }
272  }
273  }
274  if (!added && !skip) {
275  DPRINT("no default found for event %s unit mask group %d (max_grpid=%d)\n", ent->name, i, max_grpid);
276  return PFM_ERR_UMASK;
277  }
278  }
279  DPRINT("max_grpid=%d nattrs=%d k=%d umask=0x%"PRIx64"\n", max_grpid, e->nattrs, k, *umask);
280 done:
281  e->nattrs = k;
282  return PFM_SUCCESS;
283 }

365 static int
366 pfm_intel_x86_encode_gen(void *this, pfmlib_event_desc_t *e)
367 
368 {
369  pfmlib_pmu_t *pmu = this;
370  pfmlib_event_attr_info_t *a;
371  const intel_x86_entry_t *pe;
372  pfm_intel_x86_reg_t reg, reg2;
373  unsigned int grpmsk, ugrpmsk = 0;
374  uint64_t umask1, umask2, ucode, last_ucode = ~0ULL;
375  unsigned int modhw = 0;
376  unsigned int plmmsk = 0;
377  int umodmsk = 0, modmsk_r = 0;
378  int k, ret, id;
379  int max_req_grpid = -1;
380  unsigned short grpid;
381  unsigned short max_grpid = INTEL_X86_MAX_GRPID;
382  unsigned short last_grpid = INTEL_X86_MAX_GRPID;
383  unsigned short req_grpid;
384  unsigned int ldlat = 0, ldlat_um = 0;
385  unsigned int fe_thr= 0, fe_thr_um = 0;
386  int excl_grp_but_0 = -1;
387  int grpcounts[INTEL_X86_NUM_GRP];
388  int req_grps[INTEL_X86_NUM_GRP];
389  int ncombo[INTEL_X86_NUM_GRP];
390 
391  memset(grpcounts, 0, sizeof(grpcounts));
392  memset(ncombo, 0, sizeof(ncombo));
393 
394  pe = this_pe(this);
395 
396  e->fstr[0] = '\0';
397 
398  /*
399  * preset certain fields from event code
400  * including modifiers
401  */
402  reg.val = pe[e->event].code;
403 
404  grpmsk = (1 << pe[e->event].ngrp)-1;
405 
406  /* take into account hardcoded umask */
407  umask1 = (reg.val >> 8) & 0xff;
408  umask2 = 0;
409 
410  modmsk_r = pe[e->event].modmsk_req;
411 
412  for (k = 0; k < e->nattrs; k++) {
413  a = attr(e, k);
414 
415  if (a->ctrl != PFM_ATTR_CTRL_PMU)
416  continue;
417 
418  if (a->type == PFM_ATTR_UMASK) {
419  grpid = get_grpid(pe[e->event].umasks[a->idx].grpid);
420  req_grpid = get_req_grpid(pe[e->event].umasks[a->idx].grpid);
421 
422  /*
423  * certain event groups are meant to be
424  * exclusive, i.e., only unit masks of one group
425  * can be used
426  */
427  if (last_grpid != INTEL_X86_MAX_GRPID && grpid != last_grpid
428  && intel_x86_eflag(this, e->event, INTEL_X86_GRP_EXCL)) {
429  DPRINT("exclusive unit mask group error\n");
430  return PFM_ERR_FEATCOMB;
431  }
432  /*
433  * selecting certain umasks in a group may exclude any umasks
434  * from any groups with a higher index
435  *
436  * enforcement requires looking at the grpid of all the umasks
437  */
438  if (intel_x86_uflag(this, e->event, a->idx, INTEL_X86_EXCL_GRP_GT))
439  max_grpid = grpid;
440 
441  if (intel_x86_uflag(this, e->event, a->idx, INTEL_X86_EXCL_GRP_BUT_0))
442  excl_grp_but_0 = grpid;
443  /*
444  * upper layer has removed duplicates
445  * so if we come here more than once, it is for two
446  * disinct umasks
447  *
448  * NCOMBO=no combination of unit masks within the same
449  * umask group
450  */
451  ++grpcounts[grpid];
452  if (intel_x86_uflag(this, e->event, a->idx, INTEL_X86_GRP_REQ)) {
453  DPRINT("event requires grpid %d\n", req_grpid);
454  /* initialize req_grpcounts array only when needed */
455  if (max_req_grpid == -1) {
456  int x;
457  for (x = 0; x < INTEL_X86_NUM_GRP; x++)
458  req_grps[x] = 0xff;
459  }
460  if (req_grpid > max_req_grpid)
461  max_req_grpid = req_grpid;
462  DPRINT("max_req_grpid=%d\n", max_req_grpid);
463  req_grps[req_grpid] = 1;
464  }
465 
466  /* mark that we have a umask with NCOMBO in this group */
467  if (intel_x86_uflag(this, e->event, a->idx, INTEL_X86_NCOMBO))
468  ncombo[grpid] = 1;
469 
470  if (intel_x86_uflag(this, e->event, a->idx, INTEL_X86_LDLAT))
471  ldlat_um = 1;
472 
473  if (intel_x86_uflag(this, e->event, a->idx, INTEL_X86_FETHR))
474  fe_thr_um = 1;
475 
476  /*
477  * if more than one umask in this group but one is marked
478  * with ncombo, then fail. It is okay to combine umask within
479  * a group as long as none is tagged with NCOMBO
480  */
481  if (grpcounts[grpid] > 1 && ncombo[grpid]) {
482  DPRINT("umask %s does not support unit mask combination within group %d\n", pe[e->event].umasks[a->idx].uname, grpid);
483  return PFM_ERR_FEATCOMB;
484  }
485 
486  last_grpid= grpid;
487  ucode = pe[e->event].umasks[a->idx].ucode;
488  modhw |= pe[e->event].umasks[a->idx].modhw;
489  umask2 |= ucode >> 8;
490  ugrpmsk |= 1 << pe[e->event].umasks[a->idx].grpid;
491 
492  modmsk_r |= pe[e->event].umasks[a->idx].umodmsk_req;
493 
494  if (intel_x86_uflag(this, e->event, a->idx, INTEL_X86_CODE_OVERRIDE)) {
495  if (last_ucode != ~0ULL && (ucode & 0xff) != last_ucode) {
496  DPRINT("cannot override event with two different codes for %s\n", pe[e->event].name);
497  return PFM_ERR_FEATCOMB;
498  }
499  last_ucode = ucode & 0xff;
500  reg.sel_event_select = last_ucode;
501  }
502  } else if (a->type == PFM_ATTR_RAW_UMASK) {
503  int ofr_bits = 8;
504  uint64_t rmask;
505 
506  /* set limit on width of raw umask */
507  if (intel_x86_eflag(this, e->event, INTEL_X86_NHM_OFFCORE)) {
508  ofr_bits = 38;
509  if (e->pmu->pmu == PFM_PMU_INTEL_WSM || e->pmu->pmu == PFM_PMU_INTEL_WSM_DP)
510  ofr_bits = 16;
511  }
512  rmask = (1ULL << ofr_bits) - 1;
513 
514  if (a->idx & ~rmask) {
515  DPRINT("raw umask is too wide max %d bits\n", ofr_bits);
516  return PFM_ERR_ATTR;
517  }
518 
519  /* override umask */
520  umask2 = a->idx & rmask;
521  ugrpmsk = grpmsk;
522  } else {
523  uint64_t ival = e->attrs[k].ival;
524  switch(a->idx) {
525  case INTEL_X86_ATTR_I: /* invert */
526  reg.sel_inv = !!ival;
527  umodmsk |= _INTEL_X86_ATTR_I;
528  break;
529  case INTEL_X86_ATTR_E: /* edge */
530  reg.sel_edge = !!ival;
531  umodmsk |= _INTEL_X86_ATTR_E;
532  break;
533  case INTEL_X86_ATTR_C: /* counter-mask */
534  if (ival > 255)
535  return PFM_ERR_ATTR_VAL;
536  reg.sel_cnt_mask = ival;
537  umodmsk |= _INTEL_X86_ATTR_C;
538  break;
539  case INTEL_X86_ATTR_U: /* USR */
540  reg.sel_usr = !!ival;
541  plmmsk |= _INTEL_X86_ATTR_U;
542  umodmsk |= _INTEL_X86_ATTR_U;
543  break;
544  case INTEL_X86_ATTR_K: /* OS */
545  reg.sel_os = !!ival;
546  plmmsk |= _INTEL_X86_ATTR_K;
547  umodmsk |= _INTEL_X86_ATTR_K;
548  break;
549  case INTEL_X86_ATTR_T: /* anythread (v3 and above) */
550  reg.sel_anythr = !!ival;
551  umodmsk |= _INTEL_X86_ATTR_T;
552  break;
553  case INTEL_X86_ATTR_LDLAT: /* load latency */
554  if (ival < 3 || ival > 65535)
555  return PFM_ERR_ATTR_VAL;
556  ldlat = ival;
557  break;
558  case INTEL_X86_ATTR_INTX: /* in_tx */
559  reg.sel_intx = !!ival;
560  umodmsk |= _INTEL_X86_ATTR_INTX;
561  break;
562  case INTEL_X86_ATTR_INTXCP: /* in_tx_cp */
563  reg.sel_intxcp = !!ival;
564  umodmsk |= _INTEL_X86_ATTR_INTXCP;
565  break;
566  case INTEL_X86_ATTR_FETHR: /* precise frontend latency threshold */
567  if (ival < 1 || ival > 4095)
568  return PFM_ERR_ATTR_VAL;
569  fe_thr = ival;
570  umodmsk |= _INTEL_X86_ATTR_FETHR;
571  break;
572  }
573  }
574  }
575  /* check required groups are in place */
576  if (max_req_grpid != -1) {
577  int x;
578  for (x = 0; x <= max_req_grpid; x++) {
579  if (req_grps[x] == 0xff)
580  continue;
581  if ((ugrpmsk & (1 << x)) == 0) {
582  DPRINT("required grpid %d umask missing\n", x);
583  return PFM_ERR_FEATCOMB;
584  }
585  }
586  }
587 
588  /*
589  * we need to wait until all the attributes have been parsed to check
590  * for conflicts between hardcoded attributes and user-provided attributes.
591  * we do not want to depend on the order in which they are specified
592  *
593  * The test check for conflicts. It is okay to specify an attribute if
594  * it encodes to the same same value as the hardcoded value. That allows
595  * use to prase a FQESTR (fully-qualified event string) as returned by
596  * the library
597  */
598  reg2.val = (umask1 | umask2) << 8;
599  if (mdhw(modhw, umodmsk, ATTR_I) && reg2.sel_inv != reg.sel_inv)
600  return PFM_ERR_ATTR_SET;
601  if (mdhw(modhw, umodmsk, ATTR_E) && reg2.sel_edge != reg.sel_edge)
602  return PFM_ERR_ATTR_SET;
603  if (mdhw(modhw, umodmsk, ATTR_C) && reg2.sel_cnt_mask != reg.sel_cnt_mask)
604  return PFM_ERR_ATTR_SET;
605  if (mdhw(modhw, umodmsk, ATTR_U) && reg2.sel_usr != reg.sel_usr)
606  return PFM_ERR_ATTR_SET;
607  if (mdhw(modhw, umodmsk, ATTR_K) && reg2.sel_os != reg.sel_os)
608  return PFM_ERR_ATTR_SET;
609  if (mdhw(modhw, umodmsk, ATTR_T) && reg2.sel_anythr != reg.sel_anythr)
610  return PFM_ERR_ATTR_SET;
611  if (mdhw(modhw, umodmsk, ATTR_INTX) && reg2.sel_intx != reg.sel_intx)
612  return PFM_ERR_ATTR_SET;
613  if (mdhw(modhw, umodmsk, ATTR_INTXCP) && reg2.sel_intxcp != reg.sel_intxcp)
614  return PFM_ERR_ATTR_SET;
615 
616  /*
617  * handle case where no priv level mask was passed.
618  * then we use the dfl_plm
619  */
620  if (!(plmmsk & (_INTEL_X86_ATTR_K|_INTEL_X86_ATTR_U))) {
621  if ((e->dfl_plm & PFM_PLM0) && (pmu->supported_plm & PFM_PLM0))
622  reg.sel_os = 1;
623  if ((e->dfl_plm & PFM_PLM3) && (pmu->supported_plm & PFM_PLM3))
624  reg.sel_usr = 1;
625  }
626  /*
627  * check that there is at least of unit mask in each unit
628  * mask group
629  */
630  if ((ugrpmsk != grpmsk && !intel_x86_eflag(this, e->event, INTEL_X86_GRP_EXCL)) || ugrpmsk == 0) {
631  ugrpmsk ^= grpmsk;
632  ret = pfm_intel_x86_add_defaults(this, e, ugrpmsk, &umask2, max_grpid, excl_grp_but_0);
633  if (ret != PFM_SUCCESS)
634  return ret;
635  }
636  /*
637  * GRP_EXCL_BUT_0 groups require at least one bit set in grpid = 0 and one in theirs
638  * applies to OFFCORE_RESPONSE umasks on some processors (e.g., Goldmont)
639  */
640  DPRINT("excl_grp_but_0=%d\n", excl_grp_but_0);
641  if (excl_grp_but_0 != -1) {
642  /* skip group 0, because it is authorized */
643  for (k = 1; k < INTEL_X86_NUM_GRP; k++) {
644  DPRINT("grpcounts[%d]=%d\n", k, grpcounts[k]);
645  if (grpcounts[k] && k != excl_grp_but_0) {
646  DPRINT("GRP_EXCL_BUT_0 but grpcounts[%d]=%d\n", k, grpcounts[k]);
647  return PFM_ERR_FEATCOMB;
648  }
649  }
650  }
651  ret = intel_x86_check_pebs(this, e);
652  if (ret != PFM_SUCCESS)
653  return ret;
654 
655  /*
656  * check no umask violates the max_grpid constraint
657  */
658  if (max_grpid != INTEL_X86_MAX_GRPID) {
659  ret = intel_x86_check_max_grpid(this, e, max_grpid);
660  if (ret != PFM_SUCCESS) {
661  DPRINT("event %s: umask from grp > %d\n", pe[e->event].name, max_grpid);
662  return ret;
663  }
664  }
665 
666  if (modmsk_r && (umodmsk ^ modmsk_r)) {
667  DPRINT("required modifiers missing: 0x%x\n", modmsk_r);
668  return PFM_ERR_ATTR;
669  }
670  /*
671  * reorder all the attributes such that the fstr appears always
672  * the same regardless of how the attributes were submitted.
673  */
674  evt_strcat(e->fstr, "%s", pe[e->event].name);
675  pfmlib_sort_attr(e);
676  for(k=0; k < e->nattrs; k++) {
677  a = attr(e, k);
678  if (a->ctrl != PFM_ATTR_CTRL_PMU)
679  continue;
680  if (a->type == PFM_ATTR_UMASK)
681  evt_strcat(e->fstr, ":%s", pe[e->event].umasks[a->idx].uname);
682  else if (a->type == PFM_ATTR_RAW_UMASK)
683  evt_strcat(e->fstr, ":0x%x", a->idx);
684  }
685 
686  if (intel_x86_eflag(this, e->event, INTEL_X86_FRONTEND)) {
687  uint64_t um_thr = (umask2 >> 8) & 0xfff; /* threshold from umask */
688 
689  DPRINT("um_thr=0x%"PRIx64 " fe_thr=%u thr_um=%u modhw=0x%x umodhw=0x%x\n", um_thr, fe_thr, fe_thr_um, modhw, umodmsk);
690  /* umask expects a fe_thres modifier */
691  if (fe_thr_um) {
692  /* hardware has non zero fe_thres (hardcoded) */
693  if (um_thr) {
694  /* user passed fe_thres, then must match hardcoded */
695  if (mdhw(modhw, umodmsk, ATTR_FETHR)) {
696  if (fe_thr != um_thr)
697  return PFM_ERR_ATTR_SET;
698  } else
699  fe_thr = um_thr;
700  } else if (fe_thr == 0) {
701  fe_thr = INTEL_X86_FETHR_DEFAULT;
702  }
703  umask2 &= ~((0xfffULL) << 8);
704  umask2 |= fe_thr << 8;
705  }
706  }
707 
708  /*
709  * offcore_response or precise frontend require a separate register
710  */
711  if (intel_x86_eflag(this, e->event, INTEL_X86_NHM_OFFCORE)
712  || intel_x86_eflag(this, e->event, INTEL_X86_FRONTEND)) {
713  e->codes[1] = umask2;
714  e->count = 2;
715  umask2 = 0;
716  } else {
717  e->count = 1;
718  }
719 
720  if (ldlat && !ldlat_um) {
721  DPRINT("passed ldlat= but not using ldlat umask\n");
722  return PFM_ERR_ATTR;
723  }
724 
725  /*
726  * force a default ldlat (will not appear in display_reg)
727  */
728  if (ldlat_um && !ldlat) {
729  DPRINT("missing ldlat= for umask, forcing to default %d cycles\n", INTEL_X86_LDLAT_DEFAULT);
730  ldlat = INTEL_X86_LDLAT_DEFAULT;
731  }
732 
733  if (ldlat && ldlat_um) {
734  e->codes[1] = ldlat;
735  e->count = 2;
736  }
737 
738  /* take into account hardcoded modifiers, so use or on reg.val */
739  reg.val |= (umask1 | umask2) << 8;
740 
741  reg.sel_en = 1; /* force enable bit to 1 */
742  reg.sel_int = 1; /* force APIC int to 1 */
743 
744  e->codes[0] = reg.val;
745 
746  /*
747  * on recent processors (except Atom), edge requires cmask >=1
748  */
749  if ((pmu->flags & INTEL_X86_PMU_FL_ECMASK)
750  && reg.sel_edge && !reg.sel_cnt_mask) {
751  DPRINT("edge requires cmask >= 1\n");
752  return PFM_ERR_ATTR;
753  }
754 
755  /*
756  * decode ALL modifiers
757  */
758  for (k = 0; k < e->npattrs; k++) {
759  if (e->pattrs[k].ctrl != PFM_ATTR_CTRL_PMU)
760  continue;
761 
762  if (e->pattrs[k].type == PFM_ATTR_UMASK)
763  continue;
764 
765  id = e->pattrs[k].idx;
766  switch(id) {
767  case INTEL_X86_ATTR_U:
768  evt_strcat(e->fstr, ":%s=%lu", intel_x86_mods[id].name, reg.sel_usr);
769  break;
770  case INTEL_X86_ATTR_K:
771  evt_strcat(e->fstr, ":%s=%lu", intel_x86_mods[id].name, reg.sel_os);
772  break;
773  case INTEL_X86_ATTR_E:
774  evt_strcat(e->fstr, ":%s=%lu", intel_x86_mods[id].name, reg.sel_edge);
775  break;
776  case INTEL_X86_ATTR_I:
777  evt_strcat(e->fstr, ":%s=%lu", intel_x86_mods[id].name, reg.sel_inv);
778  break;
779  case INTEL_X86_ATTR_C:
780  evt_strcat(e->fstr, ":%s=%lu", intel_x86_mods[id].name, reg.sel_cnt_mask);
781  break;
782  case INTEL_X86_ATTR_T:
783  evt_strcat(e->fstr, ":%s=%lu", intel_x86_mods[id].name, reg.sel_anythr);
784  break;
785  case INTEL_X86_ATTR_LDLAT:
786  evt_strcat(e->fstr, ":%s=%d", intel_x86_mods[id].name, ldlat);
787  break;
788  case INTEL_X86_ATTR_INTX:
789  evt_strcat(e->fstr, ":%s=%lu", intel_x86_mods[id].name, reg.sel_intx);
790  break;
791  case INTEL_X86_ATTR_INTXCP:
792  evt_strcat(e->fstr, ":%s=%lu", intel_x86_mods[id].name, reg.sel_intxcp);
793  break;
794  case INTEL_X86_ATTR_FETHR:
795  evt_strcat(e->fstr, ":%s=%lu", intel_x86_mods[id].name, fe_thr);
796  break;
797  }
798  }
799  return PFM_SUCCESS;
800 }

851 int
852 pfm_intel_x86_validate_table(void *this, FILE *fp)
853 {
854  pfmlib_pmu_t *pmu = this;
855  const intel_x86_entry_t *pe = this_pe(this);
856  int ndfl[INTEL_X86_NUM_GRP];
857  int i, j, error = 0;
858  unsigned int u, v;
859  int npebs;
860 
861  if (!pmu->atdesc) {
862  fprintf(fp, "pmu: %s missing attr_desc\n", pmu->name);
863  error++;
864  }
865 
866  if (!pmu->supported_plm && pmu->type == PFM_PMU_TYPE_CORE) {
867  fprintf(fp, "pmu: %s supported_plm not set\n", pmu->name);
868  error++;
869  }
870 
871  for(i=0; i < pmu->pme_count; i++) {
872 
873  if (!is_model_event(this, i))
874  continue;
875 
876  if (!pe[i].name) {
877  fprintf(fp, "pmu: %s event%d: :: no name (prev event was %s)\n", pmu->name, i,
878  i > 1 ? pe[i-1].name : "??");
879  error++;
880  }
881 
882  if (!pe[i].desc) {
883  fprintf(fp, "pmu: %s event%d: %s :: no description\n", pmu->name, i, pe[i].name);
884  error++;
885  }
886  if (pe[i].desc && strlen(pe[i].desc) == 0) {
887  fprintf(fp, "pmu: %s event%d: %s :: empty description\n", pmu->name, i, pe[i].name);
888  error++;
889  }
890  if (!pe[i].cntmsk) {
891  fprintf(fp, "pmu: %s event%d: %s :: cntmsk=0\n", pmu->name, i, pe[i].name);
892  error++;
893  }
894 
895  if (pe[i].numasks && pe[i].ngrp == 0) {
896  fprintf(fp, "pmu: %s event%d: %s :: ngrp cannot be zero\n", pmu->name, i, pe[i].name);
897  error++;
898  }
899 
900  if (pe[i].numasks && pe[i].umasks == NULL) {
901  fprintf(fp, "pmu: %s event%d: %s :: numasks but no umasks\n", pmu->name, i, pe[i].name);
902  error++;
903  }
904 
905  if (pe[i].numasks == 0 && pe[i].umasks) {
906  fprintf(fp, "pmu: %s event%d: %s :: numasks=0 but umasks defined\n", pmu->name, i, pe[i].name);
907  error++;
908  }
909 
910  if (pe[i].numasks == 0 && pe[i].ngrp) {
911  fprintf(fp, "pmu: %s event%d: %s :: ngrp must be zero\n", pmu->name, i, pe[i].name);
912  error++;
913  }
914 
915  if (pe[i].ngrp >= INTEL_X86_NUM_GRP) {
916  fprintf(fp, "pmu: %s event%d: %s :: ngrp too big (max=%d)\n", pmu->name, i, pe[i].name, INTEL_X86_NUM_GRP);
917  error++;
918  }
919  if (pe[i].model >= PFM_PMU_MAX) {
920  fprintf(fp, "pmu: %s event%d: %s :: model too big (max=%d)\n", pmu->name, i, pe[i].name, PFM_PMU_MAX);
921  error++;
922  }
923 
924  for (j=i+1; j < (int)pmu->pme_count; j++) {
925  if (pe[i].code == pe[j].code && pe[i].model == pe[j].model && !(pe[j].equiv || pe[i].equiv) && pe[j].cntmsk == pe[i].cntmsk) {
926  fprintf(fp, "pmu: %s events %s and %s have the same code 0x%x\n", pmu->name, pe[i].name, pe[j].name, pe[i].code);
927  error++;
928  }
929  }
930 
931  for(j=0; j < INTEL_X86_NUM_GRP; j++)
932  ndfl[j] = 0;
933 
934  for(j=0, npebs = 0; j < (int)pe[i].numasks; j++) {
935 
936  if (!pe[i].umasks[j].uname) {
937  fprintf(fp, "pmu: %s event%d: %s umask%d :: no name\n", pmu->name, i, pe[i].name, j);
938  error++;
939  }
940  if (pe[i].umasks[j].modhw && (pe[i].umasks[j].modhw | pe[i].modmsk) != pe[i].modmsk) {
941  fprintf(fp, "pmu: %s event%d: %s umask%d: %s :: modhw not subset of modmsk\n", pmu->name, i, pe[i].name, j, pe[i].umasks[j].uname);
942  error++;
943  }
944 
945  if (!pe[i].umasks[j].udesc) {
946  fprintf(fp, "pmu: %s event%d: umask%d: %s :: no description\n", pmu->name, i, j, pe[i].umasks[j].uname);
947  error++;
948  }
949  if (pe[i].umasks[j].udesc && strlen(pe[i].umasks[j].udesc) == 0) {
950  fprintf(fp, "pmu: %s event%d: umask%d: %s :: empty description\n", pmu->name, i, j, pe[i].umasks[j].uname);
951  error++;
952  }
953  if (pe[i].ngrp && get_grpid(pe[i].umasks[j].grpid) >= pe[i].ngrp) {
954  fprintf(fp, "pmu: %s event%d: %s umask%d: %s :: invalid grpid %d (must be < %d)\n", pmu->name, i, pe[i].name, j, pe[i].umasks[j].uname, get_grpid(pe[i].umasks[j].grpid), pe[i].ngrp);
955  error++;
956  }
957  if (pe[i].ngrp && get_req_grpid(pe[i].umasks[j].grpid) >= pe[i].ngrp) {
958  fprintf(fp, "pmu: %s event%d: %s umask%d: %s :: invalid req_grpid %d (must be < %d)\n", pmu->name, i, pe[i].name, j, pe[i].umasks[j].uname, get_req_grpid(pe[i].umasks[j].grpid), pe[i].ngrp);
959  error++;
960  }
961  if (pe[i].umasks[j].umodel >= PFM_PMU_MAX) {
962  fprintf(fp, "pmu: %s event%d: %s umask%d: %s :: model too big (max=%d)\n", pmu->name, i, pe[i].name, j, pe[i].umasks[j].uname, PFM_PMU_MAX);
963  error++;
964  }
965  if (pe[i].umasks[j].uflags & INTEL_X86_DFL)
966  ndfl[pe[i].umasks[j].grpid]++;
967 
968  if (pe[i].umasks[j].uflags & INTEL_X86_PEBS)
969  npebs++;
970  }
971 
972  if (npebs && !intel_x86_eflag(this, i, INTEL_X86_PEBS)) {
973  fprintf(fp, "pmu: %s event%d: %s, pebs umasks but event pebs flag is not set\n", pmu->name, i, pe[i].name);
974  error++;
975  }
976 
977  if (intel_x86_eflag(this, i, INTEL_X86_PEBS) && pe[i].numasks && npebs == 0) {
978  fprintf(fp, "pmu: %s event%d: %s, pebs event flag but no umask has the pebs flag\n", pmu->name, i, pe[i].name);
979  error++;
980  }
981 
982  /* if only one umask, then ought to be default */
983  if (pe[i].numasks == 1 && !(pe[i].umasks[0].uflags & INTEL_X86_DFL)) {
984  fprintf(fp, "pmu: %s event%d: %s, only one umask but no default set\n", pmu->name, i, pe[i].name);
985  error++;
986  }
987 
988  if (pe[i].numasks) {
989  unsigned int *dfl_model = malloc(sizeof(*dfl_model) * pe[i].numasks);
990  if (!dfl_model)
991  goto skip_dfl;
992  for(u=0; u < pe[i].ngrp; u++) {
993  int l = 0, m;
994  for (v = 0; v < pe[i].numasks; v++) {
995  if (pe[i].umasks[v].grpid != u)
996  continue;
997  if (pe[i].umasks[v].uflags & INTEL_X86_DFL) {
998  for (m = 0; m < l; m++) {
999  if (dfl_model[m] == pe[i].umasks[v].umodel || dfl_model[m] == 0) {
1000  fprintf(fp, "pmu: %s event%d: %s grpid %d has 2 default umasks\n", pmu->name, i, pe[i].name, u);
1001  error++;
1002  }
1003  }
1004  if (m == l)
1005  dfl_model[l++] = pe[i].umasks[v].umodel;
1006  }
1007  }
1008  }
1009  free(dfl_model);
1010  }
1011 skip_dfl:
1012 
1013  if (pe[i].flags & INTEL_X86_NCOMBO) {
1014  fprintf(fp, "pmu: %s event%d: %s :: NCOMBO is a umask only flag\n", pmu->name, i, pe[i].name);
1015  error++;
1016  }
1017 
1018  for(u=0; u < pe[i].numasks; u++) {
1019 
1020  if (pe[i].umasks[u].uequiv)
1021  continue;
1022 
1023  if (pe[i].umasks[u].uflags & INTEL_X86_NCOMBO)
1024  continue;
1025 
1026  for(v=j+1; v < pe[i].numasks; v++) {
1027  if (pe[i].umasks[v].uequiv)
1028  continue;
1029  if (pe[i].umasks[v].uflags & INTEL_X86_NCOMBO)
1030  continue;
1031  if (pe[i].umasks[v].grpid != pe[i].umasks[u].grpid)
1032  continue;
1033  if ((pe[i].umasks[u].ucode & pe[i].umasks[v].ucode) && pe[i].umasks[u].umodel == pe[i].umasks[v].umodel) {
1034  fprintf(fp, "pmu: %s event%d: %s :: umask %s and %s have overlapping code bits\n", pmu->name, i, pe[i].name, pe[i].umasks[u].uname, pe[i].umasks[v].uname);
1035  error++;
1036  }
1037  }
1038  }
1039  }
1040  return error ? PFM_ERR_INVAL : PFM_SUCCESS;
1041 }

1043 int
1044 pfm_intel_x86_get_event_attr_info(void *this, int pidx, int attr_idx, pfmlib_event_attr_info_t *info)
1045 {
1046  const intel_x86_entry_t *pe = this_pe(this);
1047  const pfmlib_attr_desc_t *atdesc = this_atdesc(this);
1048  pfmlib_pmu_t *pmu = this;
1049  int numasks, idx;
1050 
1051  if (!is_model_event(this, pidx)) {
1052  DPRINT("invalid event index %d\n", pidx);
1053  return PFM_ERR_INVAL;
1054  }
1055 
1056  numasks = intel_x86_num_umasks(this, pidx);
1057  if (attr_idx < numasks) {
1058  int has_extpebs = pmu->flags & INTEL_X86_PMU_FL_EXTPEBS;
1059 
1060  idx = intel_x86_attr2umask(this, pidx, attr_idx);
1061  info->name = pe[pidx].umasks[idx].uname;
1062  info->desc = pe[pidx].umasks[idx].udesc;
1063  info->equiv= pe[pidx].umasks[idx].uequiv;
1064 
1065  info->code = pe[pidx].umasks[idx].ucode;
1066  if (!intel_x86_uflag(this, pidx, idx, INTEL_X86_CODE_OVERRIDE))
1067  info->code >>= 8;
1068 
1069  info->type = PFM_ATTR_UMASK;
1070  info->is_dfl = intel_x86_uflag(this, pidx, idx, INTEL_X86_DFL);
1071  info->is_precise = intel_x86_uflag(this, pidx, idx, INTEL_X86_PEBS);
1072  /*
1073  * if PEBS is supported, then hw buffer sampling is also supported
1074  * because PEBS is a hw buffer
1075  */
1076  info->support_hw_smpl = (info->is_precise || has_extpebs);
1077  /*
1078  * On Intel X86, either all or none of the umasks are speculative
1079  * for a speculative event, so propagate speculation info to all
1080  * umasks
1081  */
1082  if (pmu->flags & PFMLIB_PMU_FL_SPEC) {
1083  int ret = intel_x86_eflag(this, pidx, INTEL_X86_SPEC);
1084  if (ret)
1085  info->is_speculative = PFM_EVENT_INFO_SPEC_TRUE;
1086  else
1087  info->is_speculative = PFM_EVENT_INFO_SPEC_FALSE;
1088  } else
