315 static inline unsigned int
316 intel_x86_num_umasks(void *this, int pidx)
317 {
318  pfmlib_pmu_t *pmu = this;
319  const intel_x86_entry_t *pe = this_pe(this);
320  unsigned int i, n = 0;
321  unsigned short model;
322 
323  /*
324  * some umasks may be model specific
325  */
326  for (i = 0; i < pe[pidx].numasks; i++) {
327  model = pe[pidx].umasks[i].umodel;
328  if (model && model != pmu->pmu)
329  continue;
330  n++;
331  }
332  return n;
333 }
